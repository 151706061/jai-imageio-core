045package com.github.jaiimageio.impl.plugins.pnm;
046
047import java.util.Hashtable;
048
049import javax.imageio.ImageTypeSpecifier;
050import javax.imageio.metadata.IIOMetadataFormatImpl;
051
052public class PNMMetadataFormat extends IIOMetadataFormatImpl {
053    /** The table to link the child to its parent.
054     */
055    private static Hashtable parents = new Hashtable();
056
057    static {
058        //children for the root
059        parents.put("FormatName", "com_sun_media_imageio_plugins_pnm_image_1.0");
060        parents.put("Variant", "com_sun_media_imageio_plugins_pnm_image_1.0");
061        parents.put("Width", "com_sun_media_imageio_plugins_pnm_image_1.0");
062        parents.put("Height", "com_sun_media_imageio_plugins_pnm_image_1.0");
063        parents.put("MaximumSample", "com_sun_media_imageio_plugins_pnm_image_1.0");
064        parents.put("Comment", "com_sun_media_imageio_plugins_pnm_image_1.0");
065    }
066
067    private static PNMMetadataFormat instance;
068
069    public static synchronized PNMMetadataFormat getInstance() {
070        if (instance == null)
071            instance = new PNMMetadataFormat();
072        return instance;
073    }
074
075    String resourceBaseName = this.getClass().getName() + "Resources";
076
077    /** Constructs <code>PNMMetadataFormat</code>.  Calls the super
078     *  class constructor.  Sets the resource base name.  Adds the elements
079     *  into this format object based on the XML schema and DTD.
080     */
081    PNMMetadataFormat() {
082        super("com_sun_media_imageio_plugins_pnm_image_1.0", CHILD_POLICY_ALL);
083        setResourceBaseName(resourceBaseName);
084        addElements();
085    }
086
087    /** Adds the elements into this format object based on the XML
088     *  schema and DTD.
089     */
090    private void addElements() {
091        addElement("FormatName",
092                      getParent("FormatName"),
093                      CHILD_POLICY_EMPTY);
094
095        addElement("Variant",
096                      getParent("Variant"),
097                      CHILD_POLICY_EMPTY);
098        addElement("Width",
099                      getParent("Width"),
100                      CHILD_POLICY_EMPTY);
101        addElement("Height",
102                      getParent("Height"),
103                      CHILD_POLICY_EMPTY);
104        addElement("MaximumSample",
105                      getParent("MaximumSample"),
106                      CHILD_POLICY_EMPTY);
107        addElement("Comment",
108                      getParent("Comment"),
109                      CHILD_POLICY_EMPTY);
110    }
111
112    public String getParent(String elementName) {
113        return (String)parents.get(elementName);
114    }
115
116    public boolean canNodeAppear(String elementName,
117                                 ImageTypeSpecifier imageType) {
118        if (getParent(elementName) != null)
119            return true;
120        return false;
121    }
122}
