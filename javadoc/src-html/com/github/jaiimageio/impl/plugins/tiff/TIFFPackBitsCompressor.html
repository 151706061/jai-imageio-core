045package com.github.jaiimageio.impl.plugins.tiff;
046
047import java.io.IOException;
048
049import com.github.jaiimageio.plugins.tiff.BaselineTIFFTagSet;
050import com.github.jaiimageio.plugins.tiff.TIFFCompressor;
051
052/**
053 *
054 */
055public class TIFFPackBitsCompressor extends TIFFCompressor {
056
057    public TIFFPackBitsCompressor() {
058        super("PackBits", BaselineTIFFTagSet.COMPRESSION_PACKBITS, true);
059    }
060
061    /**
062     * Performs PackBits compression for a single buffer of data.
063     * This should be called for each row of each tile. The returned
064     * value is the offset into the output buffer after compression.
065     */
066    private static int packBits(byte[] input, int inOffset, int inCount,
067                                byte[] output, int outOffset) {
068        int inMax = inOffset + inCount - 1;
069        int inMaxMinus1 = inMax - 1;
070
071        while(inOffset <= inMax) {
072            int run = 1;
073            byte replicate = input[inOffset];
074            while(run < 127 && inOffset < inMax &&
075                  input[inOffset] == input[inOffset+1]) {
076                run++;
077                inOffset++;
078            }
079            if(run > 1) {
080                inOffset++;
081                output[outOffset++] = (byte)(-(run - 1));
082                output[outOffset++] = replicate;
083            }
084
085            run = 0;
086            int saveOffset = outOffset;
087            while(run < 128 &&
088                  ((inOffset < inMax &&
089                    input[inOffset] != input[inOffset+1]) ||
090                   (inOffset < inMaxMinus1 &&
091                    input[inOffset] != input[inOffset+2]))) {
092                run++;
093                output[++outOffset] = input[inOffset++];
094            }
095            if(run > 0) {
096                output[saveOffset] = (byte)(run - 1);
097                outOffset++;
098            }
099
100            if(inOffset == inMax) {
101                if(run > 0 && run < 128) {
102                    output[saveOffset]++;
103                    output[outOffset++] = input[inOffset++];
104                } else {
105                    output[outOffset++] = (byte)0;
106                    output[outOffset++] = input[inOffset++];
107                }
108            }
109        }
110
111        return outOffset;
112    }
113
114    public int encode(byte[] b, int off,
115                      int width, int height,
116                      int[] bitsPerSample,
117                      int scanlineStride) throws IOException {
118        int bitsPerPixel = 0;
119        for (int i = 0; i < bitsPerSample.length; i++) {
120            bitsPerPixel += bitsPerSample[i];
121        }
122        int bytesPerRow = (bitsPerPixel*width + 7)/8;
123        int bufSize = (int)(bytesPerRow + (bytesPerRow + 127)/128);
124        byte[] compData = new byte[bufSize];
125
126        int bytesWritten = 0;
127
128        for(int i = 0; i < height; i++) {
129            int bytes = packBits(b, off, scanlineStride, compData, 0);
130            off += scanlineStride;
131            bytesWritten += bytes;
132            stream.write(compData, 0, bytes);
133        }
134
135        return bytesWritten;
136    }
137}
