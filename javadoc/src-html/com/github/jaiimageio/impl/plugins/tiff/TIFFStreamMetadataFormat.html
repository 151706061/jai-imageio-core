045package com.github.jaiimageio.impl.plugins.tiff;
046
047import javax.imageio.ImageTypeSpecifier;
048import javax.imageio.metadata.IIOMetadataFormat;
049
050public class TIFFStreamMetadataFormat extends TIFFMetadataFormat {
051
052    private static TIFFStreamMetadataFormat theInstance = null;
053
054    public boolean canNodeAppear(String elementName,
055                                 ImageTypeSpecifier imageType) {
056        return false;
057    }
058
059    private TIFFStreamMetadataFormat() {
060        this.resourceBaseName =
061    "com.github.jaiimageio.impl.plugins.tiff.TIFFStreamMetadataFormatResources";
062        this.rootName = TIFFStreamMetadata.nativeMetadataFormatName;
063
064        TIFFElementInfo einfo;
065        TIFFAttrInfo ainfo;
066        String[] empty = new String[0];
067        String[] childNames;
068        String[] attrNames;
069
070        childNames = new String[] { "ByteOrder" };
071        einfo = new TIFFElementInfo(childNames, empty, CHILD_POLICY_ALL);
072
073        elementInfoMap.put(TIFFStreamMetadata.nativeMetadataFormatName,
074                           einfo);
075
076        childNames = empty;
077        attrNames = new String[] { "value" };
078        einfo = new TIFFElementInfo(childNames, attrNames, CHILD_POLICY_EMPTY);
079        elementInfoMap.put("ByteOrder", einfo);
080
081        ainfo = new TIFFAttrInfo();
082        ainfo.dataType = DATATYPE_STRING;
083        ainfo.isRequired = true;
084        attrInfoMap.put("ByteOrder/value", ainfo);
085    }
086
087    public static synchronized IIOMetadataFormat getInstance() {
088        if (theInstance == null) {
089            theInstance = new TIFFStreamMetadataFormat();
090        }
091        return theInstance;
092    }
093}
