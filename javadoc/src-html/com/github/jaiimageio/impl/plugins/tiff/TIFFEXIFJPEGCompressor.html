045package com.github.jaiimageio.impl.plugins.tiff;
046
047import javax.imageio.ImageWriteParam;
048import javax.imageio.metadata.IIOMetadata;
049
050import com.github.jaiimageio.plugins.tiff.BaselineTIFFTagSet;
051
052/**
053 * A <code>TIFFCompressor</code> for the JPEG variant of EXIF.
054 */
055public class TIFFEXIFJPEGCompressor extends TIFFBaseJPEGCompressor {
056    public TIFFEXIFJPEGCompressor(ImageWriteParam param) {
057        super(TIFFImageWriter.EXIF_JPEG_COMPRESSION_TYPE,
058              BaselineTIFFTagSet.COMPRESSION_OLD_JPEG,
059              false,
060              param);
061    }
062
063    public void setMetadata(IIOMetadata metadata) {
064        // Set the metadata.
065        super.setMetadata(metadata);
066
067        // Initialize the JPEG writer and writeparam.
068        initJPEGWriter(false, // No stream metadata (not writing abbreviated)
069                       true); // Yes image metadata (remove APPn markers)
070    }
071}
