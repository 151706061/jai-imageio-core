045package com.github.jaiimageio.impl.plugins.tiff;
046
047import com.github.jaiimageio.plugins.tiff.BaselineTIFFTagSet;
048import com.github.jaiimageio.plugins.tiff.TIFFColorConverter;
049import com.github.jaiimageio.plugins.tiff.TIFFField;
050
051/**
052 */
053public class TIFFYCbCrColorConverter extends TIFFColorConverter {
054
055    private float LumaRed = 0.299f;
056    private float LumaGreen = 0.587f;
057    private float LumaBlue = 0.114f;
058
059    private float referenceBlackY = 0.0f;
060    private float referenceWhiteY = 255.0f;
061
062    private float referenceBlackCb = 128.0f;
063    private float referenceWhiteCb = 255.0f;
064
065    private float referenceBlackCr = 128.0f;
066    private float referenceWhiteCr = 255.0f;
067
068    private float codingRangeY = 255.0f;
069    private float codingRangeCbCr = 127.0f;
070
071    public TIFFYCbCrColorConverter(TIFFImageMetadata metadata) {
072        TIFFImageMetadata tmetadata = (TIFFImageMetadata)metadata;
073
074        TIFFField f =
075           tmetadata.getTIFFField(BaselineTIFFTagSet.TAG_Y_CB_CR_COEFFICIENTS);
076        if (f != null && f.getCount() == 3) {
077            this.LumaRed = f.getAsFloat(0);
078            this.LumaGreen = f.getAsFloat(1);
079            this.LumaBlue = f.getAsFloat(2);
080        }
081
082        f =
083          tmetadata.getTIFFField(BaselineTIFFTagSet.TAG_REFERENCE_BLACK_WHITE);
084        if (f != null && f.getCount() == 6) {
085            this.referenceBlackY = f.getAsFloat(0);
086            this.referenceWhiteY = f.getAsFloat(1);
087            this.referenceBlackCb = f.getAsFloat(2);
088            this.referenceWhiteCb = f.getAsFloat(3);
089            this.referenceBlackCr = f.getAsFloat(4);
090            this.referenceWhiteCr = f.getAsFloat(5);
091        }
092    }
093
094    /*
095      The full range component value is converted from the code by:
096
097      FullRangeValue = (code - ReferenceBlack) * CodingRange
098                / (ReferenceWhite - ReferenceBlack);
099
100      The code is converted from the full-range component value by:
101
102      code = (FullRangeValue * (ReferenceWhite - ReferenceBlack)
103                / CodingRange) + ReferenceBlack;
104
105     */
106    public void fromRGB(float r, float g, float b, float[] result) {
107        // Convert RGB to full-range YCbCr.
108        float Y = (LumaRed*r + LumaGreen*g + LumaBlue*b);
109        float Cb = (b - Y)/(2 - 2*LumaBlue);
110        float Cr = (r - Y)/(2 - 2*LumaRed);
111
112        // Convert full-range YCbCr to code.
113        result[0] = Y*(referenceWhiteY - referenceBlackY)/codingRangeY +
114            referenceBlackY;
115        result[1] = Cb*(referenceWhiteCb - referenceBlackCb)/codingRangeCbCr +
116            referenceBlackCb;
117        result[2] = Cr*(referenceWhiteCr - referenceBlackCr)/codingRangeCbCr +
118            referenceBlackCr;
119    }
120
121    public void toRGB(float x0, float x1, float x2, float[] rgb) {
122        // Convert YCbCr code to full-range YCbCr.
123        float Y = (x0 - referenceBlackY)*codingRangeY/
124            (referenceWhiteY - referenceBlackY);
125        float Cb = (x1 - referenceBlackCb)*codingRangeCbCr/
126            (referenceWhiteCb - referenceBlackCb);
127        float Cr = (x2 - referenceBlackCr)*codingRangeCbCr/
128            (referenceWhiteCr - referenceBlackCr);
129
130        // Convert YCbCr to RGB.
131        rgb[0] = Cr*(2 - 2*LumaRed) + Y;
132        rgb[2] = Cb*(2 - 2*LumaBlue) + Y;
133        rgb[1] = (Y - LumaBlue*rgb[2] - LumaRed*rgb[0])/LumaGreen;
134    }
135}
