045package com.github.jaiimageio.impl.plugins.bmp;
046
047import java.awt.image.DataBuffer;
048import java.awt.image.SampleModel;
049import java.awt.image.SinglePixelPackedSampleModel;
050import java.util.Locale;
051
052import javax.imageio.IIOException;
053import javax.imageio.ImageTypeSpecifier;
054import javax.imageio.ImageWriter;
055import javax.imageio.spi.ImageWriterSpi;
056import javax.imageio.spi.ServiceRegistry;
057
058import com.github.jaiimageio.impl.common.ImageUtil;
059import com.github.jaiimageio.impl.common.PackageUtil;
060
061public class BMPImageWriterSpi extends ImageWriterSpi {
062    private static String [] readerSpiNames =
063        {"com.github.jaiimageio.impl.plugins.bmp.BMPImageReaderSpi"};
064    private static String[] formatNames = {"bmp", "BMP"};
065    private static String[] extensions = {"bmp"};
066    private static String[] mimeTypes = {
067        "image/bmp", "image/x-bmp", "image/x-windows-bmp"
068    };
069    private boolean registered = false;
070
071    public BMPImageWriterSpi() {
072        super(PackageUtil.getVendor(),
073              PackageUtil.getVersion(),
074              formatNames,
075              extensions,
076              mimeTypes,
077              "com.github.jaiimageio.impl.plugins.bmp.BMPImageWriter",
078              STANDARD_OUTPUT_TYPE,
079              readerSpiNames,
080              false,
081              null, null, null, null,
082              true,
083              BMPMetadata.nativeMetadataFormatName,
084              "com.github.jaiimageio.impl.plugins.bmp.BMPMetadataFormat",
085              null, null);
086    }
087
088    public String getDescription(Locale locale) {
089        String desc = PackageUtil.getSpecificationTitle() + 
090            " BMP Image Writer";  
091        return desc;
092    }
093
094    public void onRegistration(ServiceRegistry registry,
095                               Class category) {
096        if (registered) {
097            return;
098        }
099        
100        registered = true;
101        
102        // By JDK 1.7, the BMPImageWriter will have been in JDK core for 
103        // atleast two FCS releases, so we can set JIIO's to lower priority
104        // With JDK 1.8, we can entirely de-register the JIIO one
105        ImageUtil.processOnRegistration(registry, category, "BMP", this,
106                                        8, 7); // JDK version 1.8, 1.7
107    }
108
109    public boolean canEncodeImage(ImageTypeSpecifier type) {
110        int dataType= type.getSampleModel().getDataType();
111        if (dataType < DataBuffer.TYPE_BYTE || dataType > DataBuffer.TYPE_INT)
112            return false;
113       
114        SampleModel sm = type.getSampleModel();
115        int numBands = sm.getNumBands();
116        if (!(numBands == 1 || numBands == 3))
117            return false;
118        
119        if (numBands == 1 && dataType != DataBuffer.TYPE_BYTE)
120            return false;
121        
122        if (dataType > DataBuffer.TYPE_BYTE && 
123              !(sm instanceof SinglePixelPackedSampleModel))
124            return false;
125        
126        return true;
127    }
128
129    public ImageWriter createWriterInstance(Object extension)
130        throws IIOException {
131        return new BMPImageWriter(this);
132    }
133}
