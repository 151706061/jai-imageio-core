082/**
083 * This class defines the content of the <code>ImageInputStream</code>
084 * containing several raw images with the same image type: the number of
085 * the images, the image type, the offset for the first sample of each image,
086 * and the image size information.
087 *
088 * <p><code>ImageInputStream</code> methods are not commented individually.
089 * These methods merely forward the call to the <code>ImageInputStream</code>
090 * specified when the <code>RawImageInputStream</code> is constructed.</p>
091 */
092public class RawImageInputStream implements ImageInputStream {
093    private static final String[] preDefinedColorSpaces =
094        new String[]{"GRAY", "sRGB", "LINEAR_RGB", "PYCC", "CIEXYZ"};
095
096    private static final int[]  preDefinedTypes = new int[]{
097        ColorSpace.CS_GRAY, ColorSpace.CS_sRGB, ColorSpace.CS_LINEAR_RGB,
098        ColorSpace.CS_PYCC, ColorSpace.CS_CIEXYZ
099    };
100
101    /** Gets the attribute from the node. */
102    private static String getAttribute(Node node, String name) {
103        NamedNodeMap map = node.getAttributes();
104        node = map.getNamedItem(name);
105        return (node != null) ? node.getNodeValue() : null;
106    }
107
108    /** Gets the boolean type attribute. */
109    private static boolean getBoolean(Node node, String name) {
110        String s = getAttribute(node, name);
111        return (s == null) ? false : (new Boolean(s)).booleanValue();
112    }
113
114    /** Gets the integer type attribute. */
115    private static int getInt(Node node, String name) {
116        String s = getAttribute(node, name);
117        return (s == null) ? 0 : (new Integer(s)).intValue();
118    }
119
120    /** Gets the integer type attribute. */
121    private static byte[] getByteArray(Node node, String name) {
122        String s = getAttribute(node, name);
123        if (s == null)
124            return null;
125        StringTokenizer token = new StringTokenizer(s);
126        int count = token.countTokens();
127        if (count == 0)
128            return null;
129
130        byte[] buf = new byte[count];
131        int i = 0;
132        while(token.hasMoreElements()) {
133            buf[i++] = new Byte(token.nextToken()).byteValue();
134        }
135        return buf;
136    }
137
138    /** Gets the integer type attribute. */
139    private static int[] getIntArray(Node node, String name) {
140        String s = getAttribute(node, name);
141        if (s == null)
142            return null;
143
144        StringTokenizer token = new StringTokenizer(s);
145        int count = token.countTokens();
146        if (count == 0)
147            return null;
148
149        int[] buf = new int[count];
150        int i = 0;
151        while(token.hasMoreElements()) {
152            buf[i++] = new Integer(token.nextToken()).intValue();
153        }
154        return buf;
155    }
156
157    private static int getTransparency(String s) {
158        if ("BITMASK".equals(s))
159            return Transparency.BITMASK;
160        else if ("OPAQUE".equals(s))
161            return Transparency.OPAQUE;
162        else if ("TRANSLUCENT".equals(s))
163            return Transparency.TRANSLUCENT;
164        else return 0;
165    }
166
167    private static ColorSpace getColorSpace(Node node) throws IOException {
168        NodeList nodes = node.getChildNodes();
169        for (int i = 0; i < nodes.getLength(); i++) {
170            Node child = nodes.item(i);
171            if ("colorSpace".equals(child.getNodeName())) {
172                String s = child.getNodeValue();
173                for (int j = 0; j < preDefinedColorSpaces.length; j++) {
174                    if (preDefinedColorSpaces[j].equals(s))
175                        return ColorSpace.getInstance(preDefinedTypes[j]);
176                }
177
178                InputStream stm = new URL(s).openStream();
179
180                ColorSpace cp = new ICC_ColorSpace(ICC_Profile.getInstance(stm));
181                stm.close();
182                return cp;
183            }
184        }
185        return null;
186    }

230    /**
231     * Constructor.
232     *
233     * <p>This constructor is the same as
234     * {@link #RawImageInputStream(ImageInputStream,ImageTypeSpecifier,
235     * long[],Dimension[])} except that a <code>SampleModel</code> is
236     * supplied instead of an <code>ImageTypeSpecifier</code>. This
237     * constructor creates a <code>ColorModel</code> for the supplied
238     * <code>SampleModel</code> and then creates an
239     * <code>ImageTypeSpecifier</code>.</p>
240     *
241     * <p>Suitable <code>ColorModel</code>s are guaranteed to exist
242     * for all instances of <code>ComponentSampleModel</code>.
243     * For 1- and 3- banded <code>SampleModel</code>s, the
244     * <code>ColorModel</code> will be opaque.  For 2- and 4-banded
245     * <code>SampleModel</code>s, the output will use alpha transparency
246     * which is not premultiplied.  1- and 2-banded data will use a
247     * grayscale <code>ColorSpace</code>, and 3- and 4-banded data a sRGB
248     * <code>ColorSpace</code>. Data with 5 or more bands will use a
249     * <code>ColorSpace</code> which satisfies compatibility constraints
250     * but is merely a placeholder and <i>does not perform correct color
251     * conversion to and from the C.I.E. XYZ and sRGB color spaces</i>.</p>
252     *
253     * <p>An instance of <code>DirectColorModel</code> will be created for
254     * instances of <code>SinglePixelPackedSampleModel</code> with no more
255     * than 4 bands.</p>
256     *
257     * <p>An instance of <code>IndexColorModel</code> will be created for
258     * instances of <code>MultiPixelPackedSampleModel</code>. The colormap
259     * will be a grayscale ramp with <code>1&nbsp;<<&nbsp;numberOfBits</code>
260     * entries ranging from zero to at most 255.</p>
261     *
262     *  @param source The <code>ImageInputStream</code> containing all the raw
263     *                images.
264     *  @param sampleModel The <code>SampleModel</code> for all the images
265     *                in the stream.
266     *  @param imageOffsets The position of the first sample for each image
267     *                in the stream.
268     *  @param imageDimensions The image size for each image in the stream.
269     *
270     *  @throws IllegalArgumentException If <code>sampleModel</code> is null.
271     *  @throws IllegalArgumentException If the sizes of
272     *                <code>imageOffsets</code>
273     *                and <code>imageDimensions</code> are different or if
274     *                either array is null.
275     *  @throws IllegalArgumentException If it is not possible to create a
276     *                <code>ColorModel</code> from the supplied
277     *                <code>SampleModel</code>.
278     */
279    public RawImageInputStream(ImageInputStream source,
280                               SampleModel sampleModel,
281                               long[] imageOffsets,
282                               Dimension[] imageDimensions) {
283        if (imageOffsets == null || imageDimensions == null ||
284            imageOffsets.length != imageDimensions.length) {
285            throw new IllegalArgumentException
286                (I18N.getString("RawImageInputStream0"));
287        }
288
289        this.source = source;
290        ColorModel colorModel = ImageUtil.createColorModel(sampleModel);
291        if(colorModel == null) {
292            throw new IllegalArgumentException
293                (I18N.getString("RawImageInputStream4"));
294        }
295        this.type = new ImageTypeSpecifier(colorModel, sampleModel);
296        this.imageOffsets = imageOffsets;
297        this.imageDimensions = imageDimensions;
298    }

483     *
484     *  @throws RuntimeException If the parse configuration isn't correct.
485     *
486     *  @throws IllegalArgumentException If the number of "width" elements isn't
487     *                the same as the number of "height" elements.

492     *                from a URL.

494    public RawImageInputStream(ImageInputStream source,
495                               org.xml.sax.InputSource xmlSource)
496                               throws SAXException, IOException {
497        this.source = source;
498
499        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
500        dbf.setValidating(true);
501        dbf.setNamespaceAware(true);
502        dbf.setAttribute("http://java.sun.com/xml/jaxp/properties/schemaLanguage",
503                       "http://www.w3.org/2001/XMLSchema");
504        DocumentBuilder db = null;
505        try {
506            db = dbf.newDocumentBuilder();
507        } catch (ParserConfigurationException ex) {
508            throw new RuntimeException(I18N.getString("RawImageInputStream1"),
509                                                      ex);
510        }
511
512        Document doc = db.parse(xmlSource);
513
514        //gets the byte order
515        NodeList nodes = doc.getElementsByTagName("byteOrder");
516        String byteOrder = nodes.item(0).getNodeValue();
517        if ("NETWORK".equals(byteOrder)) {
518            this.setByteOrder(ByteOrder.BIG_ENDIAN);
519            this.source.setByteOrder(ByteOrder.BIG_ENDIAN);
520        } else  if ("REVERSE".equals(byteOrder)) {
521            this.setByteOrder(ByteOrder.LITTLE_ENDIAN);
522            this.setByteOrder(ByteOrder.LITTLE_ENDIAN);
523        }
524
525        //gets image offsets
526        nodes = doc.getElementsByTagName("offset");
527        int length = nodes.getLength();
528        this.imageOffsets = new long[length];
529        for (int i = 0; i < length; i++) {
530            imageOffsets[i] = new Long(nodes.item(i).getNodeValue()).longValue();
531        }
532
533        //gets image dimensions
534        nodes = doc.getElementsByTagName("width");
535        NodeList nodes1 = doc.getElementsByTagName("height");
536        length = nodes.getLength();
537        if (length != nodes1.getLength())
538            throw new IllegalArgumentException
539                (I18N.getString("RawImageInputStream2"));
540
541        this.imageDimensions = new Dimension[length];
542        for (int i = 0; i < length; i++) {
543            String w = nodes.item(i).getNodeValue();
544            String h = nodes1.item(i).getNodeValue();
545
546            imageDimensions[i] =
547                new Dimension((new Integer(w)).intValue(),
548                              (new Integer(h)).intValue());
549        }
550
551        //get sampleModel
552        SampleModel sampleModel = null;
553
554        // for ComponentSampleModel
555        nodes = doc.getElementsByTagName("ComponentSampleModel");
556        if (nodes.getLength() > 0) {
557            Node node = nodes.item(0);
558            int[] bankIndices =getIntArray (node, "bankIndices");
559
560            if (bankIndices == null)
561                sampleModel =
562                    new ComponentSampleModel(getInt(node, "dataType"),
563                                             getInt(node, "w"),
564                                             getInt(node, "h"),
565                                             getInt(node, "pixelStride"),
566                                             getInt(node, "scanlineStride"),
567                                             getIntArray(node, "bandOffsets"));
568            else
569                sampleModel =
570                    new ComponentSampleModel(getInt(node, "dataType"),
571                                             getInt(node, "w"),
572                                             getInt(node, "h"),
573                                             getInt(node, "pixelStride"),
574                                             getInt(node, "scanlineStride"),
575                                             bankIndices,
576                                             getIntArray(node, "bandOffsets"));
577        }
578
579        // for MultiPixelPackedSampleModel
580        nodes = doc.getElementsByTagName("MultiPixelPackedSampleModel");
581        if (nodes.getLength() > 0) {
582            Node node = nodes.item(0);
583            sampleModel =
584                new MultiPixelPackedSampleModel(getInt(node, "dataType"),
585                                         getInt(node, "w"),
586                                         getInt(node, "h"),
587                                         getInt(node, "numberOfBits"),
588                                         getInt(node, "scanlineStride"),
589                                         getInt(node, "dataBitOffset"));
590        }
591
592        // for SinglePixelPackedSampleModel
593        nodes = doc.getElementsByTagName("SinglePixelPackedSampleModel");
594        if (nodes.getLength() > 0) {
595            Node node = nodes.item(0);
596            sampleModel =
597                new SinglePixelPackedSampleModel(getInt(node, "dataType"),
598                                         getInt(node, "w"),
599                                         getInt(node, "h"),
600                                         getInt(node, "scanlineStride"),
601                                         getIntArray(node, "bitMasks"));
602        }
603
604        //get colorModel
605        ColorModel colorModel = null;
606
607        // for ComponentColorModel
608        nodes = doc.getElementsByTagName("ComponentColorModel");
609        if (nodes.getLength() > 0) {
610            Node node = nodes.item(0);
611            colorModel =
612                new ComponentColorModel(getColorSpace(node),
613                                        getIntArray(node, "bits"),
614                                        getBoolean(node, "hasAlpha"),
615                                        getBoolean(node, "isAlphaPremultiplied"),
616                                        getTransparency(getAttribute(node, "transparency")),
617                                        getInt(node, "transferType"));
618        }
619
620        // for DirectColorModel
621        nodes = doc.getElementsByTagName("DirectColorModel");
622        if (nodes.getLength() > 0) {
623            Node node = nodes.item(0);
624            colorModel =
625                new DirectColorModel(getColorSpace(node),
626                                     getInt(node, "bits"),
627                                     getInt(node, "rmask"),
628                                     getInt(node, "gmask"),
629                                     getInt(node, "bmask"),
630                                     getInt(node, "amask"),
631                                     false,
632                                     Transparency.OPAQUE);
633        }
634
635        // for IndexColorModel
636        nodes = doc.getElementsByTagName("IndexColorModel");
637        if (nodes.getLength() > 0) {
638            Node node = nodes.item(0);
639            byte[] alpha = getByteArray(node, "a");
640
641            if (alpha == null)
642                colorModel =
643                    new IndexColorModel(getInt(node, "bits"),
644                                        getInt(node, "size"),
645                                        getByteArray(node, "r"),
646                                        getByteArray(node, "g"),
647                                        getByteArray(node, "b"));
648            else
649                colorModel =
650                    new IndexColorModel(getInt(node, "bits"),
651                                        getInt(node, "size"),
652                                        getByteArray(node, "r"),
653                                        getByteArray(node, "g"),
654                                        getByteArray(node, "b"),
655                                        alpha);
656        }
657
658        //create image type
659        this.type = new ImageTypeSpecifier(colorModel, sampleModel);
660
661        //assign imagedimension based on the sample model
662        if (this.imageDimensions.length == 0) {
663            this.imageDimensions = new Dimension[this.imageOffsets.length];
664
665            imageDimensions[0] = new Dimension(sampleModel.getWidth(),
666                                               sampleModel.getHeight());
667            for (int i = 1; i < imageDimensions.length; i++)
668                imageDimensions[i] = imageDimensions[0];
669        }
670    }
