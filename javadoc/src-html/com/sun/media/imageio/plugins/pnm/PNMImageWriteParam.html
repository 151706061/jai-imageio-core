045package com.sun.media.imageio.plugins.pnm;
046
047import javax.imageio.ImageWriteParam;
048
049/**
050 * A subclass of <code>ImageWriteParam</code> for writing images in
051 * the PNM format.
052 *
053 * <p> This class allows for the specification of whether to write
054 * in the ASCII or raw variants of the PBM, PGM, and PPM formats;
055 * by default, the raw variant is used.
056 */
057public class PNMImageWriteParam extends ImageWriteParam {
058
059    private boolean raw = true;
060
061    /**
062     * Constructs a <code>PNMImageWriteParam</code> object with default values
063     * for parameters.
064     */
065    public PNMImageWriteParam() {}
066
067    /**
068     * Sets the representation to be used.  If the <code>raw</code>
069     * parameter is <code>true</code>, the raw representation will be used;
070     * otherwise the ASCII representation will be used.
071     *
072     * @param raw <code>true</code> if raw format is to be used.
073     * @see #getRaw()
074     */
075    public void setRaw(boolean raw) {
076        this.raw = raw;
077    }
078
079    /**
080     * Returns the value of the <code>raw</code> parameter.  The default
081     * value is <code>true</code>.
082     *
083     * @return whether the data are written in raw representation.
084     * @see #setRaw(boolean)
085     */
086    public boolean getRaw() {
087        return raw;
088    }
089}
