045package com.sun.media.imageioimpl.common;
046
047import java.io.InputStream;
048import java.util.PropertyResourceBundle;
049
050/**
051 * Class to simplify use of internationalization message strings.
052 * Property files are constructed in terms of content as for JAI with
053 * one "key=value" pair per line. All such files however have the same
054 * name "properties". The resource extractor resolves the extraction of
055 * the file from the jar as the package name is included automatically.
056 *
057 * <p>Extenders need only provide a static method
058 * <code>getString(String)</code> which calls the static method in this
059 * class with the name of the invoking class and returns a
060 * <code>String</code>.
061 */
062public class I18NImpl {
063    /**
064     * Returns the message string with the specified key from the
065     * "properties" file in the package containing the class with
066     * the specified name.
067     */
068    protected static final String getString(String className, String key) {
069        PropertyResourceBundle bundle = null;
070        try {
071            InputStream stream =
072                Class.forName(className).getResourceAsStream("properties");
073            bundle = new PropertyResourceBundle(stream);
074        } catch(Throwable e) {
075            throw new RuntimeException(e); // Chain the exception.
076        }
077
078        return (String)bundle.handleGetObject(key);
079    }
080}
