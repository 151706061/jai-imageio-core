046package com.sun.media.imageioimpl.plugins.wbmp;
047
048import java.io.UnsupportedEncodingException;
049import java.util.ArrayList;
050import java.util.Iterator;
051import java.util.List;
052import javax.imageio.ImageTypeSpecifier;
053import javax.imageio.metadata.IIOMetadata;
054import javax.imageio.metadata.IIOMetadataNode;
055import javax.imageio.metadata.IIOMetadataFormat;
056import javax.imageio.metadata.IIOMetadataFormatImpl;
057import org.w3c.dom.Node;
058import com.sun.media.imageioimpl.common.ImageUtil;
059
060public class WBMPMetadata extends IIOMetadata {
061
062    public static final String nativeMetadataFormatName =
063        "com_sun_media_imageio_plugins_wbmp_image_1.0";
064
065    public int wbmpType;
066    
067    public int width;
068    public int height;
069
070    public WBMPMetadata() {
071        super(true,
072              nativeMetadataFormatName,
073              "com.sun.media.imageioimpl.plugins.wbmp.WBMPMetadataFormat",
074              null, null);
075    }
076
077    public boolean isReadOnly() {
078        return true;
079    }
080    
081    public Node getAsTree(String formatName) {
082        if (formatName.equals(nativeMetadataFormatName)) {
083            return getNativeTree();
084        } else if (formatName.equals
085                   (IIOMetadataFormatImpl.standardMetadataFormatName)) {
086            return getStandardTree();
087        } else {
088            throw new IllegalArgumentException(I18N.getString("WBMPMetadata0"));
089        }
090    }
091
092    private Node getNativeTree() {
093        IIOMetadataNode root =
094            new IIOMetadataNode(nativeMetadataFormatName);
095
096        addChildNode(root, "WBMPType", new Integer(wbmpType));
097        addChildNode(root, "Width", new Integer(width));
098        addChildNode(root, "Height", new Integer(height));
099        
100        return root;
101    }
102
103    public void setFromTree(String formatName, Node root) {
104        throw new IllegalStateException(I18N.getString("WBMPMetadata1"));
105    }
106
107    public void mergeTree(String formatName, Node root) {
108        throw new IllegalStateException(I18N.getString("WBMPMetadata1"));
109    }
110
111    public void reset() {
112        throw new IllegalStateException(I18N.getString("WBMPMetadata1"));
113    }
114
115    private IIOMetadataNode addChildNode(IIOMetadataNode root,
116                                         String name,
117                                         Object object) {
118        IIOMetadataNode child = new IIOMetadataNode(name);
119        if (object != null) {
120            child.setUserObject(object);
121            child.setNodeValue(ImageUtil.convertObjectToString(object));
122        }
123        root.appendChild(child);
124        return child;
125    }
126    
127    
128    protected IIOMetadataNode getStandardChromaNode() {
129
130        IIOMetadataNode node = new IIOMetadataNode("Chroma");
131
132        IIOMetadataNode subNode = new IIOMetadataNode("ColorSpaceType");
133        subNode.setAttribute("name", "GRAY");
134        node.appendChild(subNode);
135
136        subNode = new IIOMetadataNode("NumChannels");
137        subNode.setAttribute("value", "1");
138        node.appendChild(subNode);
139
140        subNode = new IIOMetadataNode("BlackIsZero");
141        subNode.setAttribute("value", "TRUE");
142        node.appendChild(subNode);
143
144        return node;
145    }
146
147    
148    protected IIOMetadataNode getStandardDataNode() {
149        IIOMetadataNode node = new IIOMetadataNode("Data");
150
151        IIOMetadataNode subNode = new IIOMetadataNode("SampleFormat");
152        subNode.setAttribute("value", "UnsignedIntegral");
153        node.appendChild(subNode);
154
155        subNode = new IIOMetadataNode("BitsPerSample");
156        subNode.setAttribute("value", "1");
157        node.appendChild(subNode);
158
159        return node;
160    }
161
162    protected IIOMetadataNode getStandardDimensionNode() {
163        IIOMetadataNode dimension_node = new IIOMetadataNode("Dimension");
164        IIOMetadataNode node = null; // scratch node
165
166        // PixelAspectRatio not in image
167
168        node = new IIOMetadataNode("ImageOrientation");
169        node.setAttribute("value", "Normal");
170        dimension_node.appendChild(node);
171
172        return dimension_node;
173    }
174
175}
