281    private static IIOImage getThumbnail(ImageInputStream stream, int len,
282                                         int thumbnailType, int w, int h)
283        throws IOException {
284
285        IIOImage result;
286
287        long startPos = stream.getStreamPosition();
288
289        if(thumbnailType == THUMBNAIL_JPEG) {
290            Iterator readers = ImageIO.getImageReaders(stream);
291            if(readers == null || !readers.hasNext()) return null;
292            ImageReader reader = (ImageReader)readers.next();
293            reader.setInput(stream);
294            BufferedImage image = reader.read(0, null);
295            IIOMetadata metadata = null;
296            try {
297                metadata = reader.getImageMetadata(0);
298            } catch(Exception e) {
299                // Ignore it
300            }
301            result = new IIOImage(image, null, metadata);
302        } else {
303            int numBands;
304            ColorModel cm;
305            if(thumbnailType == THUMBNAIL_PALETTE) {
306                if(len < 768 + w*h) {
307                    return null;
308                }
309
310                numBands = 1;
311
312                byte[] palette = new byte[768];
313                stream.readFully(palette);
314                byte[] r = new byte[256];
315                byte[] g = new byte[256];
316                byte[] b = new byte[256];
317                for(int i = 0, off = 0; i < 256; i++) {
318                    r[i] = palette[off++];
319                    g[i] = palette[off++];
320                    b[i] = palette[off++];
321                }
322
323                cm = new IndexColorModel(8, 256, r, g, b);
324            } else {
325                if(len < 3*w*h) {
326                    return null;
327                }
328
329                numBands = 3;
330
331                ColorSpace cs = ColorSpace.getInstance(ColorSpace.CS_sRGB);
332                cm = new ComponentColorModel(cs, false, false,
333                                             Transparency.OPAQUE,
334                                             DataBuffer.TYPE_BYTE);
335            }
336
337            byte[] data = new byte[w*h*numBands];
338            stream.readFully(data);
339            DataBufferByte db = new DataBufferByte(data, data.length);
340            WritableRaster wr =
341                Raster.createInterleavedRaster(db, w, h, w*numBands, numBands,
342                                               new int[] {0, 1, 2}, null);
343            BufferedImage image = new BufferedImage(cm, wr, false, null);
344            result = new IIOImage(image, null, null);
345        }
346
347        stream.seek(startPos + len);
348
349        return result;
350    }

459    private class QTable {
460        private static final int QTABLE_SIZE = 64;
461
462        int elementPrecision;
463        int tableID;
464        JPEGQTable table;
465
466        int length;
467
468        QTable(ImageInputStream stream) throws IOException {
469            elementPrecision = (int)stream.readBits(4);
470            tableID = (int)stream.readBits(4);
471            byte[] tmp = new byte[QTABLE_SIZE];
472            stream.readFully(tmp);
473            int[] data = new int[QTABLE_SIZE];
474            for (int i = 0; i < QTABLE_SIZE; i++) {
475                data[i] = tmp[zigzag[i]] & 0xff;
476            }
477            table = new JPEGQTable(data);
478            length = data.length + 1;
479        }
480    }

482    private class HuffmanTable {
483        private static final int NUM_LENGTHS = 16;
484
485        int tableClass;
486        int tableID;
487        JPEGHuffmanTable table;
488
489        int length;
490
491        HuffmanTable(ImageInputStream stream) throws IOException {
492            tableClass = (int)stream.readBits(4);
493            tableID = (int)stream.readBits(4);
494            short[] lengths = new short[NUM_LENGTHS];
495            for (int i = 0; i < NUM_LENGTHS; i++) {
496                lengths[i] = (short)stream.read();
497            }
498            int numValues = 0;
499            for (int i = 0; i < NUM_LENGTHS; i++) {
500                numValues += lengths[i];
501            }
502            short[] values = new short[numValues];
503            for (int i = 0; i < numValues; i++) {
504                values[i] = (short)stream.read();
505            }
506            table = new JPEGHuffmanTable(lengths, values);
507
508            length = 1 + NUM_LENGTHS + values.length;
509        }
510    }

512    private synchronized void initializeFromStream(ImageInputStream iis)
513        throws IOException {
514        iis.mark();
515        iis.setByteOrder(ByteOrder.BIG_ENDIAN);
516
517        markers = new ArrayList();
518
519        boolean isICCProfileValid = true;
520        int numICCProfileChunks = 0;
521        long[] iccProfileChunkOffsets = null;
522        int[] iccProfileChunkLengths = null;
523
524        while(true) {
525            try {
526                // 0xff denotes a potential marker.
527                if(iis.read() == 0xff) {
528                    // Get next byte.
529                    int code = iis.read();
530
531                    // Is a marker if and only if code not in {0x00, 0xff}.
532                    // Continue to next marker if this is not a marker or if
533                    // it is an empty marker.
534                    if(code == 0x00 || code == 0xff ||
535                       code == SOI || code == TEM ||
536                       (code >= RST_MIN && code <= RST_MAX)) {
537                        continue;
538                    }
539
540                    // If at the end, quit.
541                    if(code == EOI) {
542                        break;
543                    }
544
545                    // Get the content length.
546                    int dataLength = iis.readUnsignedShort() - 2;
547
548                    if(APPN_MIN <= code && code <= APPN_MAX) {
549                        long pos = iis.getStreamPosition();
550                        boolean appnAdded = false;
551
552                        switch(code) {
553                        case APP0:
554                            if(dataLength >= 5) {
555                                byte[] b = new byte[5];
556                                iis.readFully(b);
557                                String id = new String(b);
558                                if(id.startsWith("JFIF") &&
559                                   !app0JFIFPresent) {
560                                    app0JFIFPresent = true;
561                                    markers.add(new Integer(APP0_JFIF));
562                                    majorVersion = iis.read();
563                                    minorVersion = iis.read();
564                                    resUnits = iis.read();
565                                    Xdensity = iis.readUnsignedShort();
566                                    Ydensity = iis.readUnsignedShort();
567                                    thumbWidth = iis.read();
568                                    thumbHeight = iis.read();
569                                    if(thumbWidth > 0 && thumbHeight > 0) {
570                                        IIOImage imiio =
571                                            getThumbnail(iis, dataLength - 14,
572                                                         THUMBNAIL_RGB,
573                                                         thumbWidth,
574                                                         thumbHeight);
575                                        if(imiio != null) {
576                                            jfifThumbnail = (BufferedImage)
577                                                imiio.getRenderedImage();
578                                        }
579                                    }
580                                    appnAdded = true;
581                                } else if(id.startsWith("JFXX")) {
582                                    if(!app0JFXXPresent) {
583                                        extensionCodes = new ArrayList(1);
584                                        jfxxThumbnails = new ArrayList(1);
585                                        app0JFXXPresent = true;
586                                    }
587                                    markers.add(new Integer(APP0_JFXX));
588                                    int extCode = iis.read();
589                                    extensionCodes.add(new Integer(extCode));
590                                    int w = 0, h = 0, offset = 6;
591                                    if(extCode != THUMBNAIL_JPEG) {
592                                        w = iis.read();
593                                        h = iis.read();
594                                        offset += 2;
595                                    }
596                                    IIOImage imiio =
597                                        getThumbnail(iis, dataLength - offset,
598                                                     extCode, w, h);
599                                    if(imiio != null) {
600                                        jfxxThumbnails.add(imiio);
601                                    }
602                                    appnAdded = true;
603                                }
604                            }
605                            break;
606                        case APP1:
607                            if(dataLength >= 6) {
608                                byte[] b = new byte[6];
609                                iis.readFully(b);
610                                if(b[0] == (byte)'E' &&
611                                   b[1] == (byte)'x' &&
612                                   b[2] == (byte)'i' &&
613                                   b[3] == (byte)'f' &&
614                                   b[4] == (byte)0 &&
615                                   b[5] == (byte)0) {
616                                    exifData = new byte[dataLength - 6];
617                                    iis.readFully(exifData);
618                                }
619                            }
620                        case APP2:
621                            if(dataLength >= 12) {
622                                byte[] b = new byte[12];
623                                iis.readFully(b);
624                                String id = new String(b);
625                                if(id.startsWith("ICC_PROFILE")) {
626                                    if(!isICCProfileValid) {
627                                        iis.skipBytes(dataLength - 12);
628                                        continue;
629                                    }
630
631                                    int chunkNum = iis.read();
632                                    int numChunks = iis.read();
633                                    if(numChunks == 0 ||
634                                       chunkNum == 0 ||
635                                       chunkNum > numChunks ||
636                                       (app2ICCPresent &&
637                                        (numChunks != numICCProfileChunks ||
638                                         iccProfileChunkOffsets[chunkNum]
639                                         != 0L))) {
640                                        isICCProfileValid = false;
641                                        iis.skipBytes(dataLength - 14);
642                                        continue;
643                                    }
644
645                                    if(!app2ICCPresent) {
646                                        app2ICCPresent = true;
647                                        // Only flag one marker even though
648                                        // multiple may be present.
649                                        markers.add(new Integer(APP2_ICC));
650
651                                        numICCProfileChunks = numChunks;
652
653                                        if(numChunks == 1) {
654                                            b = new byte[dataLength - 14];
655                                            iis.readFully(b);
656                                            profile =
657                                                ICC_Profile.getInstance(b);
658                                        } else {
659                                            iccProfileChunkOffsets =
660                                                new long[numChunks + 1];
661                                            iccProfileChunkLengths =
662                                                new int[numChunks + 1];
663                                            iccProfileChunkOffsets[chunkNum] =
664                                                iis.getStreamPosition();
665                                            iccProfileChunkLengths[chunkNum] =
666                                                dataLength - 14;
667                                            iis.skipBytes(dataLength - 14);
668                                        }
669                                    } else {
670                                        iccProfileChunkOffsets[chunkNum] =
671                                            iis.getStreamPosition();
672                                        iccProfileChunkLengths[chunkNum] =
673                                            dataLength - 14;
674                                        iis.skipBytes(dataLength - 14);
675                                    }
676
677                                    appnAdded = true;
678                                }
679                            }
680                            break;
681                        case APP14:
682                            if(dataLength >= 5) {
683                                byte[] b = new byte[5];
684                                iis.readFully(b);
685                                String id = new String(b);
686                                if(id.startsWith("Adobe") &&
687                                   !app14AdobePresent) { // Adobe segment
688                                    app14AdobePresent = true;
689                                    markers.add(new Integer(APP14_ADOBE));
690                                    version = iis.readUnsignedShort();
691                                    flags0 = iis.readUnsignedShort();
692                                    flags1 = iis.readUnsignedShort();
693                                    transform = iis.read();
694                                    iis.skipBytes(dataLength - 12);
695                                    appnAdded = true;
696                                }
697                            }
698                            break;
699                        default:
700                            appnAdded = false;
701                            break;
702                        }
703
704                        if(!appnAdded) {
705                            iis.seek(pos);
706                            addUnknownMarkerSegment(iis, code, dataLength);
707                        }
708                    } else if(code == DQT) {
709                        if(!dqtPresent) {
710                            dqtPresent = true;
711                            qtables = new ArrayList(1);
712                        }
713                        markers.add(new Integer(DQT));
714                        List l = new ArrayList(1);
715                        do {
716                            QTable t = new QTable(iis);
717                            l.add(t);
718                            dataLength -= t.length;
719                        } while(dataLength > 0);
720                        qtables.add(l);
721                    } else if(code == DHT) {
722                        if(!dhtPresent) {
723                            dhtPresent = true;
724                            htables = new ArrayList(1);
725                        }
726                        markers.add(new Integer(DHT));
727                        List l = new ArrayList(1);
728                        do {
729                            HuffmanTable t = new HuffmanTable(iis);
730                            l.add(t);
731                            dataLength -= t.length;
732                        } while(dataLength > 0);
733                        htables.add(l);
734                    } else if(code == DRI) {
735                        if(!driPresent) {
736                            driPresent = true;
737                        }
738                        markers.add(new Integer(DRI));
739                        driInterval = iis.readUnsignedShort();
740                    } else if(code == COM) {
741                        if(!comPresent) {
742                            comPresent = true;
743                            comments = new ArrayList(1);
744                        }
745                        markers.add(new Integer(COM));
746                        byte[] b = new byte[dataLength];
747                        iis.readFully(b);
748                        comments.add(b);
749                    } else if((code >= SOFN_MIN && code <= SOFN_MAX) ||
750                              code == SOF55) { // SOFn
751                        if(!sofPresent) {
752                            sofPresent = true;
753                            sofProcess = code - SOFN_MIN;
754                            samplePrecision = iis.read();
755                            numLines = iis.readUnsignedShort();
756                            samplesPerLine = iis.readUnsignedShort();
757                            numFrameComponents = iis.read();
758                            componentId = new int[numFrameComponents];
759                            hSamplingFactor = new int[numFrameComponents];
760                            vSamplingFactor = new int[numFrameComponents];
761                            qtableSelector = new int[numFrameComponents];
762                            for(int i = 0; i < numFrameComponents; i++) {
763                                componentId[i] = iis.read();
764                                hSamplingFactor[i] = (int)iis.readBits(4);
765                                vSamplingFactor[i] = (int)iis.readBits(4);
766                                qtableSelector[i] = iis.read();
767                            }
768                            markers.add(new Integer(SOF_MARKER));
769                        }
770                    } else if(code == SOS) {
771                        if(!sosPresent) {
772                            sosPresent = true;
773                            numScanComponents = iis.read();
774                            componentSelector = new int[numScanComponents];
775                            dcHuffTable = new int[numScanComponents];
776                            acHuffTable = new int[numScanComponents];
777                            for(int i = 0; i < numScanComponents; i++) {
778                                componentSelector[i] = iis.read();
779                                dcHuffTable[i] = (int)iis.readBits(4);
780                                acHuffTable[i] = (int)iis.readBits(4);
781                            }
782                            startSpectralSelection = iis.read();
783                            endSpectralSelection = iis.read();
784                            approxHigh = (int)iis.readBits(4);
785                            approxLow = (int)iis.readBits(4);
786                            markers.add(new Integer(SOS));
787                        }
788                        break;
789                    } else { // Any other marker
790                        addUnknownMarkerSegment(iis, code, dataLength);
791                    }
792                }
793            } catch(EOFException eofe) {
794                // XXX Should this be caught?
795                break;
796            }
797        }
798
799        if(app2ICCPresent && isICCProfileValid && profile == null) {
800            int profileDataLength = 0;
801            for(int i = 1; i <= numICCProfileChunks; i++) {
802                if(iccProfileChunkOffsets[i] == 0L) {
803                    isICCProfileValid = false;
804                    break;
805                }
806                profileDataLength += iccProfileChunkLengths[i];
807            }
808
809            if(isICCProfileValid) {
810                byte[] b = new byte[profileDataLength];
811                int off = 0;
812                for(int i = 1; i <= numICCProfileChunks; i++) {
813                    iis.seek(iccProfileChunkOffsets[i]);
814                    iis.read(b, off, iccProfileChunkLengths[i]);
815                    off += iccProfileChunkLengths[i];
816                }
817
818                profile = ICC_Profile.getInstance(b);
819            }
820        }
821
822        iis.reset();
823    }

937                    // No JFIFthumb node will be appended.
938                }

941                    if(extensionCode.intValue() == THUMBNAIL_JPEG) {

945                                thumbMetadata.getAsTree(nativeMetadataFormatName);

962                                               Integer.toString(thumb.getHeight()));

1090    protected IIOMetadataNode getStandardChromaNode() {
1091        if(!sofPresent) {
1092            // No image, so no chroma
1093            return null;
1094        }
1095
1096        IIOMetadataNode chroma = new IIOMetadataNode("Chroma");
1097        IIOMetadataNode csType = new IIOMetadataNode("ColorSpaceType");
1098        chroma.appendChild(csType);
1099
1100        IIOMetadataNode numChanNode = new IIOMetadataNode("NumChannels");
1101        chroma.appendChild(numChanNode);
1102        numChanNode.setAttribute("value",
1103                                 Integer.toString(numFrameComponents));
1104
1105        // Check JFIF presence.
1106        if(app0JFIFPresent) {
1107            if(numFrameComponents == 1) {
1108                csType.setAttribute("name", "GRAY");
1109            } else {
1110                csType.setAttribute("name", "YCbCr");
1111            }
1112            return chroma;
1113        }
1114
1115        // How about an Adobe marker segment?
1116        if(app14AdobePresent){
1117            switch(transform) {
1118            case ADOBE_TRANSFORM_YCCK: // YCCK
1119                csType.setAttribute("name", "YCCK");
1120                break;
1121            case ADOBE_TRANSFORM_YCC: // YCC
1122                csType.setAttribute("name", "YCbCr");
1123                break;
1124            case ADOBE_TRANSFORM_UNKNOWN: // Unknown
1125                if(numFrameComponents == 3) {
1126                    csType.setAttribute("name", "RGB");
1127                } else if(numFrameComponents == 4) {
1128                    csType.setAttribute("name", "CMYK");
1129                }
1130                break;
1131            }
1132            return chroma;
1133        }
1134
1135        // Initially assume no opacity.
1136        hasAlpha = false;
1137
1138        // Neither marker.  Check components
1139        if(numFrameComponents < 3) {
1140            csType.setAttribute("name", "GRAY");
1141            if(numFrameComponents == 2) {
1142                hasAlpha = true;
1143            }
1144            return chroma;
1145        }
1146
1147        boolean idsAreJFIF = true;
1148
1149        for(int i = 0; i < componentId.length; i++) {
1150            int id = componentId[i];
1151            if((id < 1) || (id >= componentId.length)) {
1152                idsAreJFIF = false;
1153            }
1154        }
1155        
1156        if(idsAreJFIF) {
1157            csType.setAttribute("name", "YCbCr");
1158            if(numFrameComponents == 4) {
1159                hasAlpha = true;
1160            }
1161            return chroma;
1162        }
1163
1164        // Check against the letters
1165        if(componentId[0] == 'R' &&
1166           componentId[1] == 'G' &&
1167           componentId[2] == 'B'){
1168            csType.setAttribute("name", "RGB");
1169            if(numFrameComponents == 4 && componentId[3] == 'A') {
1170                hasAlpha = true;
1171            }
1172            return chroma;
1173        }
1174        
1175        if(componentId[0] == 'Y' &&
1176           componentId[1] == 'C' &&
1177           componentId[2] == 'c'){
1178            csType.setAttribute("name", "PhotoYCC");
1179            if(numFrameComponents == 4 &&
1180               componentId[3] == 'A') {
1181                hasAlpha = true;
1182            }            
1183            return chroma;
1184        }
1185        
1186        // Finally, 3-channel subsampled are YCbCr, unsubsampled are RGB
1187        // 4-channel subsampled are YCbCrA, unsubsampled are CMYK
1188
1189        boolean subsampled = false;
1190
1191        int hfactor = hSamplingFactor[0];
1192        int vfactor = vSamplingFactor[0];
1193
1194        for(int i = 1; i < componentId.length; i++) {
1195            if(hSamplingFactor[i] != hfactor ||
1196               vSamplingFactor[i] != vfactor){
1197                subsampled = true;
1198                break;
1199            }
1200        }
1201
1202        if(subsampled) {
1203            csType.setAttribute("name", "YCbCr");
1204            if(numFrameComponents == 4) {
1205                hasAlpha = true;
1206            }
1207            return chroma;
1208        }
1209         
1210        // Not subsampled.  numFrameComponents < 3 is taken care of above
1211        if(numFrameComponents == 3) {
1212            csType.setAttribute("name", "RGB");
1213        } else {
1214            csType.setAttribute("name", "CMYK");
1215        }
1216
1217        return chroma;
1218    }

1248                    new IIOMetadataNode("NumProgressiveScans");

1258        IIOMetadataNode dim = new IIOMetadataNode("Dimension");
1259        IIOMetadataNode orient = new IIOMetadataNode("ImageOrientation");
1260        orient.setAttribute("value", "normal");

1263        if(app0JFIFPresent) {

1272            IIOMetadataNode aspect = new IIOMetadataNode("PixelAspectRatio");

1275

1277                // 1 == dpi, 2 == dpc
1278                float scale = (resUnits == JFIF_RESUNITS_DPI) ? 25.4F : 10.0F;
1279

1281                    new IIOMetadataNode("HorizontalPixelSize");
1282                horiz.setAttribute("value",

1285

1287                    new IIOMetadataNode("VerticalPixelSize");
1288                vert.setAttribute("value", 
1289                                  Float.toString(scale/Ydensity));

1292        }

1323            IIOMetadataNode alpha = new IIOMetadataNode("Alpha");
1324            alpha.setAttribute("value", "nonpremultiplied"); // Always assume
1325            trans.appendChild(alpha);
1326        }
1327        return trans;
1328    }

1331
1332    private Node getTIFFTree() {
1333        String metadataName = TIFF_FORMAT;
1334
1335        BaselineTIFFTagSet base = BaselineTIFFTagSet.getInstance();
1336
1337        TIFFDirectory dir =
1338            new TIFFDirectory(new TIFFTagSet[] {
1339                base, EXIFParentTIFFTagSet.getInstance()
1340            }, null);
1341
1342        if(sofPresent) {
1343            // sofProcess -> Compression ?
1344            int compression = BaselineTIFFTagSet.COMPRESSION_JPEG;
1345            TIFFField compressionField =
1346                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_COMPRESSION),
1347                              compression);
1348            dir.addTIFFField(compressionField);
1349
1350            // samplePrecision -> BitsPerSample
1351            char[] bitsPerSample = new char[numFrameComponents];
1352            Arrays.fill(bitsPerSample, (char)(samplePrecision & 0xff));
1353            TIFFField bitsPerSampleField =
1354                new TIFFField(
1355                              base.getTag(BaselineTIFFTagSet.TAG_BITS_PER_SAMPLE),
1356                              TIFFTag.TIFF_SHORT,
1357                              bitsPerSample.length,
1358                              bitsPerSample);
1359            dir.addTIFFField(bitsPerSampleField);
1360
1361            // numLines -> ImageLength
1362            TIFFField imageLengthField =
1363                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_IMAGE_LENGTH),
1364                              numLines);
1365            dir.addTIFFField(imageLengthField);
1366
1367            // samplesPerLine -> ImageWidth
1368            TIFFField imageWidthField =
1369                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_IMAGE_WIDTH),
1370                              samplesPerLine);
1371            dir.addTIFFField(imageWidthField);
1372
1373            // numFrameComponents -> SamplesPerPixel
1374            TIFFField samplesPerPixelField =
1375                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_SAMPLES_PER_PIXEL),
1376                              numFrameComponents);
1377            dir.addTIFFField(samplesPerPixelField);
1378
1379            // componentId -> PhotometricInterpretation + ExtraSamples
1380            IIOMetadataNode chroma = getStandardChromaNode();
1381            if(chroma != null) {
1382                IIOMetadataNode csType =
1383                    (IIOMetadataNode)chroma.getElementsByTagName("ColorSpaceType").item(0);
1384                String name = csType.getAttribute("name");
1385                int photometricInterpretation = -1;
1386                if(name.equals("GRAY")) {
1387                    photometricInterpretation =
1388                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO;
1389                } else if(name.equals("YCbCr") || name.equals("PhotoYCC")) {
1390                    // NOTE: PhotoYCC -> YCbCr
1391                    photometricInterpretation =
1392                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR;
1393                } else if(name.equals("RGB")) {
1394                    photometricInterpretation =
1395                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_RGB;
1396                } else if(name.equals("CMYK") || name.equals("YCCK")) {
1397                    // NOTE: YCCK -> CMYK
1398                    photometricInterpretation =
1399                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_CMYK;
1400                }
1401
1402                if(photometricInterpretation != -1) {
1403                    TIFFField photometricInterpretationField =
1404                        new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_PHOTOMETRIC_INTERPRETATION),
1405                                      photometricInterpretation);
1406                    dir.addTIFFField(photometricInterpretationField);
1407                }
1408
1409                if(hasAlpha) {
1410                    char[] extraSamples =
1411                        new char[] {BaselineTIFFTagSet.EXTRA_SAMPLES_ASSOCIATED_ALPHA};
1412                    TIFFField extraSamplesField =
1413                        new TIFFField(
1414                                      base.getTag(BaselineTIFFTagSet.TAG_EXTRA_SAMPLES),
1415                                      TIFFTag.TIFF_SHORT,
1416                                      extraSamples.length,
1417                                      extraSamples);
1418                    dir.addTIFFField(extraSamplesField);
1419                }
1420            } // chroma != null
1421        } // sofPresent
1422
1423        // JFIF APP0 -> Resolution fields.
1424        if(app0JFIFPresent) {
1425            long[][] xResolution = new long[][] {{Xdensity, 1}};
1426            TIFFField XResolutionField =
1427                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_X_RESOLUTION),
1428                              TIFFTag.TIFF_RATIONAL,
1429                              1,
1430                              xResolution);
1431            dir.addTIFFField(XResolutionField);
1432
1433            long[][] yResolution = new long[][] {{Ydensity, 1}};
1434            TIFFField YResolutionField =
1435                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_Y_RESOLUTION),
1436                              TIFFTag.TIFF_RATIONAL,
1437                              1,
1438                              yResolution);
1439            dir.addTIFFField(YResolutionField);
1440
1441            int resolutionUnit = BaselineTIFFTagSet.RESOLUTION_UNIT_NONE;
1442            switch(resUnits) {
1443            case JFIF_RESUNITS_ASPECT:
1444                resolutionUnit = BaselineTIFFTagSet.RESOLUTION_UNIT_NONE;
1445            case JFIF_RESUNITS_DPI:
1446                resolutionUnit = BaselineTIFFTagSet.RESOLUTION_UNIT_INCH;
1447                break;
1448            case JFIF_RESUNITS_DPC:
1449                resolutionUnit = BaselineTIFFTagSet.RESOLUTION_UNIT_CENTIMETER;
1450                break;
1451            }
1452            TIFFField ResolutionUnitField =
1453                new TIFFField(base.getTag
1454                              (BaselineTIFFTagSet.TAG_RESOLUTION_UNIT),
1455                              resolutionUnit);
1456            dir.addTIFFField(ResolutionUnitField);
1457        }
1458
1459        // DQT + DHT -> JPEGTables.
1460        byte[] jpegTablesData = null;
1461        if(dqtPresent || dqtPresent) {
1462            // Determine length of JPEGTables data.
1463            int jpegTablesLength = 2; // SOI
1464            if(dqtPresent) {
1465                Iterator dqts = qtables.iterator();
1466                while(dqts.hasNext()) {
1467                    Iterator qtiter = ((List)dqts.next()).iterator();
1468                    while(qtiter.hasNext()) {
1469                        QTable qt = (QTable)qtiter.next();
1470                        jpegTablesLength += 4 + qt.length;
1471                    }
1472                }
1473            }
1474            if(dhtPresent) {
1475                Iterator dhts = htables.iterator();
1476                while(dhts.hasNext()) {
1477                    Iterator htiter = ((List)dhts.next()).iterator();
1478                    while(htiter.hasNext()) {
1479                        HuffmanTable ht = (HuffmanTable)htiter.next();
1480                        jpegTablesLength += 4 + ht.length;
1481                    }
1482                }
1483            }
1484            jpegTablesLength += 2; // EOI
1485
1486            // Allocate space.
1487            jpegTablesData = new byte[jpegTablesLength];
1488
1489            // SOI
1490            jpegTablesData[0] = (byte)0xff;
1491            jpegTablesData[1] = (byte)SOI;
1492            int jpoff = 2;
1493
1494            if(dqtPresent) {
1495                Iterator dqts = qtables.iterator();
1496                while(dqts.hasNext()) {
1497                    Iterator qtiter = ((List)dqts.next()).iterator();
1498                    while(qtiter.hasNext()) {
1499                        jpegTablesData[jpoff++] = (byte)0xff;
1500                        jpegTablesData[jpoff++] = (byte)DQT;
1501                        QTable qt = (QTable)qtiter.next();
1502                        int qtlength = qt.length + 2;
1503                        jpegTablesData[jpoff++] =
1504                            (byte)((qtlength & 0xff00) >> 8);
1505                        jpegTablesData[jpoff++] = (byte)(qtlength & 0xff);
1506                        jpegTablesData[jpoff++] =
1507                            (byte)(((qt.elementPrecision & 0xf0) << 4) |
1508                                   (qt.tableID & 0x0f));
1509                        int[] table = qt.table.getTable();
1510                        int qlen = table.length;
1511                        for(int i = 0; i < qlen; i++) {
1512                            jpegTablesData[jpoff + zigzag[i]] = (byte)table[i];
1513                        }
1514                        jpoff += qlen;
1515                    }
1516                }
1517            }
1518
1519            if(dhtPresent) {
1520                Iterator dhts = htables.iterator();
1521                while(dhts.hasNext()) {
1522                    Iterator htiter = ((List)dhts.next()).iterator();
1523                    while(htiter.hasNext()) {
1524                        jpegTablesData[jpoff++] = (byte)0xff;
1525                        jpegTablesData[jpoff++] = (byte)DHT;
1526                        HuffmanTable ht = (HuffmanTable)htiter.next();
1527                        int htlength = ht.length + 2;
1528                        jpegTablesData[jpoff++] =
1529                            (byte)((htlength & 0xff00) >> 8);
1530                        jpegTablesData[jpoff++] = (byte)(htlength & 0xff);
1531                        jpegTablesData[jpoff++] =
1532                            (byte)(((ht.tableClass & 0x0f) << 4) |
1533                                   (ht.tableID & 0x0f));
1534                        short[] lengths = ht.table.getLengths();
1535                        int numLengths = lengths.length;
1536                        for(int i = 0; i < numLengths; i++) {
1537                            jpegTablesData[jpoff++] = (byte)lengths[i];
1538                        }
1539                        short[] values = ht.table.getValues();
1540                        int numValues = values.length;
1541                        for(int i = 0; i < numValues; i++) {
1542                            jpegTablesData[jpoff++] = (byte)values[i];
1543                        }
1544                    }
1545                }
1546            }
1547
1548            jpegTablesData[jpoff++] = (byte)0xff;
1549            jpegTablesData[jpoff] = (byte)EOI;            
1550        }
1551        if(jpegTablesData != null) {
1552            TIFFField JPEGTablesField =
1553                new TIFFField(base.getTag(BaselineTIFFTagSet.TAG_JPEG_TABLES),
1554                              TIFFTag.TIFF_UNDEFINED,
1555                              jpegTablesData.length,
1556                              jpegTablesData);
1557            dir.addTIFFField(JPEGTablesField);
1558        }
1559
1560        IIOMetadata tiffMetadata = dir.getAsMetadata();
1561
1562        if(exifData != null) {
1563            try {
1564                Iterator tiffReaders =
1565                    ImageIO.getImageReadersByFormatName("TIFF");
1566                if(tiffReaders != null && tiffReaders.hasNext()) {
1567                    ImageReader tiffReader = (ImageReader)tiffReaders.next();
1568                    ByteArrayInputStream bais =
1569                        new ByteArrayInputStream(exifData);
1570                    ImageInputStream exifStream =
1571                        new MemoryCacheImageInputStream(bais);
1572                    tiffReader.setInput(exifStream);
1573                    IIOMetadata exifMetadata = tiffReader.getImageMetadata(0);
1574                    tiffMetadata.mergeTree(metadataName,
1575                                           exifMetadata.getAsTree(metadataName));
1576                    tiffReader.reset();
1577                }
1578            } catch(IOException ioe) {
1579                // Ignore it.
1580            }
1581        }
1582
1583        return tiffMetadata.getAsTree(metadataName);
1584    }

1588    private void initializeThumbnails() {
1589        synchronized(thumbnails) {
1590            if(!thumbnailsInitialized) {
1591                // JFIF/JFXX are not supposed to coexist in the same
1592                // JPEG stream but in reality sometimes they do.
1593
1594                // JFIF thumbnail
1595                if(app0JFIFPresent && jfifThumbnail != null) {
1596                    thumbnails.add(jfifThumbnail);
1597                }
1598
1599                // JFXX thumbnail(s)
1600                if(app0JFXXPresent && jfxxThumbnails != null) {
1601                    int numJFXX = jfxxThumbnails.size();
1602                    for(int i = 0; i < numJFXX; i++) {
1603                        IIOImage img = (IIOImage)jfxxThumbnails.get(i);
1604                        BufferedImage jfxxThumbnail =
1605                            (BufferedImage)img.getRenderedImage();
1606                        thumbnails.add(jfxxThumbnail);
1607                    }
1608                }
1609
1610                // EXIF thumbnail
1611                if(exifData != null) {
1612                    try {
1613                        Iterator tiffReaders =
1614                            ImageIO.getImageReadersByFormatName("TIFF");
1615                        if(tiffReaders != null && tiffReaders.hasNext()) {
1616                            ImageReader tiffReader =
1617                                (ImageReader)tiffReaders.next();
1618                            ByteArrayInputStream bais =
1619                                new ByteArrayInputStream(exifData);
1620                            ImageInputStream exifStream =
1621                                new MemoryCacheImageInputStream(bais);
1622                            tiffReader.setInput(exifStream);
1623                            if(tiffReader.getNumImages(true) > 1) {
1624                                BufferedImage exifThumbnail =
1625                                    tiffReader.read(1, null);
1626                                thumbnails.add(exifThumbnail);
1627                            }
1628                            tiffReader.reset();
1629                        }
1630                    } catch(IOException ioe) {
1631                        // Ignore it.
1632                    }
1633                }
1634
1635                thumbnailsInitialized = true;
1636            } // if(!thumbnailsInitialized)
1637        } // sychronized
1638    }

1640    int getNumThumbnails() throws IOException {
1641        initializeThumbnails();
1642        return thumbnails.size();
1643    }

1645    BufferedImage getThumbnail(int thumbnailIndex) throws IOException {
1646        if(thumbnailIndex < 0) {
1647            throw new IndexOutOfBoundsException("thumbnailIndex < 0!");

1651

1654                ("thumbnailIndex > getNumThumbnails()");
