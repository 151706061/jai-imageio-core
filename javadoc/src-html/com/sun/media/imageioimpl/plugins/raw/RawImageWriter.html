083/**
084 * The Java Image IO plugin writer for encoding a binary RenderedImage into
085 * a Raw format.
086 *
087 * <p> The encoding process may clip, subsample or select bands using the
088 * parameters specified in the <code>ImageWriteParam</code>.
089 *
090 * Thus, when read this raw image the proper image data type
091 * should be provided.
092 *
093 * @see com.sun.media.imageio.plugins.RawImageWriteParam
094 */
095
096 // <p> If the destination data is packed packed, the data is written in the
097 // order defined by the sample model. That the data is packed is defined as
098 // (1) if the sample model is <code>SingleSamplePackedSampleModel</code> or
099 // <code>BandedSampleModel</code>; or (2) the pixel stride or sanline stride
100 // equals to the band number; or (3) the pixel stride equals to the band
101 // number multiply the tile height; or (4) the scanline stride equals to the
102 // band number multiply the tile width; or (5) the data for a band is stored
103 // in a separate data bank.
104 //
105 // <p> Otherwise, the data is reordered in a packed pixel interleaved format,
106 // and then written into the stream.  In this case the data order may be change.
107 // For example, the original image data is in the order of
108 // <pre>
109 //        RRRRRRRRRRRRRRRRRRRR
110 //        GGGGGGGGGGGGGGGGGGGG
111 //        BBBBBBBBBBBBBBBBBBBB
112 //        RRRRRRRRRRRRRRRRRRRR
113 //        GGGGGGGGGGGGGGGGGGGG
114 //        BBBBBBBBBBBBBBBBBBBB
115 // </pre>
116 //
117 // , and only the G and B bands are written in the stream.  So the data in the
118 // stream will be in the order of
119 // <pre>
120 //       GBGBGBGBGBGBGB
121 // </pre>.

217    public void write(IIOMetadata streamMetadata,
218                      IIOImage image,
219                      ImageWriteParam param) throws IOException {
220        clearAbortRequest();
221        processImageStarted(imageIndex++);
222
223        if (param == null)
224            param = getDefaultWriteParam();
225
226        writeRaster = image.hasRaster();
227        Rectangle sourceRegion = param.getSourceRegion();
228        ColorModel colorModel = null;
229        Rectangle originalRegion = null;
230
231        if (writeRaster) {
232            inputRaster = image.getRaster();
233            sampleModel = inputRaster.getSampleModel();
234            originalRegion = inputRaster.getBounds();
235        } else {
236            input = image.getRenderedImage();
237            sampleModel = input.getSampleModel();
238
239            originalRegion = new Rectangle(input.getMinX(), input.getMinY(),
240                                           input.getWidth(), input.getHeight());
241
242            colorModel = input.getColorModel();
243        }
244
245        if (sourceRegion == null)
246            sourceRegion = (Rectangle)originalRegion.clone();
247        else
248            sourceRegion = sourceRegion.intersection(originalRegion);
249
250        if (sourceRegion.isEmpty())
251            throw new RuntimeException(I18N.getString("RawImageWriter1"));
252
253        scaleX = param.getSourceXSubsampling();
254        scaleY = param.getSourceYSubsampling();
255        xOffset = param.getSubsamplingXOffset();
256        yOffset = param.getSubsamplingYOffset();
257
258        sourceRegion.translate(xOffset, yOffset);
259        sourceRegion.width -= xOffset;
260        sourceRegion.height -= yOffset;
261
262        xOffset = sourceRegion.x % scaleX;
263        yOffset = sourceRegion.y % scaleY;
264
265        int minX = sourceRegion.x / scaleX;
266        int minY = sourceRegion.y / scaleY;
267        int w = (sourceRegion.width + scaleX - 1) / scaleX;
268        int h = (sourceRegion.height + scaleY - 1) / scaleY;
269
270        destinationRegion = new Rectangle(minX, minY, w, h);
271        noTransform = destinationRegion.equals(originalRegion);
272
273        tileHeight = sampleModel.getHeight();
274        tileWidth = sampleModel.getWidth();
275        if (noTransform) {
276            if (writeRaster) {
277                tileXOffset = inputRaster.getMinX();
278                tileYOffset = inputRaster.getMinY();
279            } else {
280                tileXOffset = input.getTileGridXOffset();
281                tileYOffset = input.getTileGridYOffset();
282            }
283        } else {
284            tileXOffset = destinationRegion.x;
285            tileYOffset = destinationRegion.y;
286        }
287
288        sourceBands = param.getSourceBands();
289        boolean noSubband = true;
290        numBands = sampleModel.getNumBands();
291
292        if (sourceBands != null) {
293            sampleModel = sampleModel.createSubsetSampleModel(sourceBands);
294            colorModel = null;
295            noSubband = false;
296            numBands = sampleModel.getNumBands();
297        } else {
298            sourceBands = new int[numBands];
299            for (int i = 0; i < numBands; i++)
300                sourceBands[i] = i;
301        }
302
303        if (sampleModel instanceof ComponentSampleModel) {
304            ComponentSampleModel csm = (ComponentSampleModel)sampleModel;
305            int[] bandOffsets = csm.getBandOffsets();
306
307            bandStride = bandOffsets[0];
308
309            for (int i = 1; i < bandOffsets.length; i++)
310                if (bandStride > bandOffsets[i])
311                    bandStride = bandOffsets[i];
312
313            int[] bankIndices = csm.getBankIndices();
314            int numBank = bankIndices[0];
315            for (int i = 1; i < bankIndices.length; i++)
316                if (numBank > bankIndices[i])
317                    numBank = bankIndices[i];
318
319            pxlStride = csm.getPixelStride();
320            lineStride = csm.getScanlineStride();
321
322            optimal = bandStride == 0 ||
323                      (pxlStride < lineStride && pxlStride == numBands) ||
324                      (lineStride < pxlStride && lineStride == numBands) ||
325                      (pxlStride < lineStride &&
326                       lineStride == numBands * csm.getWidth()) ||
327                      (lineStride < pxlStride &&
328                       pxlStride == numBands * csm.getHeight()) ||
329                      csm instanceof BandedSampleModel;
330        } else if (sampleModel instanceof SinglePixelPackedSampleModel ||
331                   sampleModel instanceof MultiPixelPackedSampleModel) {
332            optimal = true;
333        }
334
335        int numXTiles = getMaxTileX() - getMinTileX() + 1;
336        int totalTiles = numXTiles * (getMaxTileY() - getMinTileY() + 1);
337
338        for (int y = getMinTileY(); y <= getMaxTileY(); y++) {
339            for (int x = getMinTileX(); x <= getMaxTileX(); x++) {
340                writeRaster(getTile(x, y));
341
342                float percentage = (x + y * numXTiles + 1.0F) / totalTiles;
343                processImageProgress(percentage * 100.0F);
344            }
345        }
346
347        stream.flush();
348        if (abortRequested())
349            processWriteAborted();
350        else
351            processImageComplete();
352    }

363    private void writeRaster(Raster raster) throws IOException {
364        int numBank = 0;
365        int bandStride = 0;
366        int[] bankIndices = null;
367        int[] bandOffsets = null;
368        int bandSize = 0;
369        int numBand = sampleModel.getNumBands();
370        int type = sampleModel.getDataType();
371
372        if (sampleModel instanceof ComponentSampleModel) {
373            ComponentSampleModel csm = (ComponentSampleModel)sampleModel;
374
375            bandOffsets = csm.getBandOffsets();
376            for (int i = 0; i < numBand; i++)
377                if (bandStride < bandOffsets[i])
378                    bandStride = bandOffsets[i];
379
380            bankIndices = csm.getBankIndices();
381            for (int i = 0; i < numBand; i++)
382                if (numBank < bankIndices[i])
383                    numBank = bankIndices[i];
384
385            bandSize = (int)ImageUtil.getBandSize(sampleModel) ;
386        }
387
388        byte[] bdata = null;
389        short[] sdata = null;
390        int[] idata = null;
391        float[] fdata = null;
392        double[] ddata = null;
393
394        if (raster.getParent() != null &&
395            !sampleModel.equals(raster.getParent().getSampleModel())) {
396            WritableRaster ras =
397                Raster.createWritableRaster(sampleModel,
398                                            new Point(raster.getMinX(),
399                                                      raster.getMinY()));
400            ras.setRect(raster);
401            raster = ras;
402        }
403
404        DataBuffer data = raster.getDataBuffer();
405
406        if (optimal) {
407            if (numBank > 0) {  //multiple data bank
408                for (int i = 0; i < numBands; i++) {
409                    int bank = bankIndices[sourceBands[i]];
410                    switch (type) {
411                    case DataBuffer.TYPE_BYTE:
412                        bdata = ((DataBufferByte)data).getData(bank);
413                        stream.write(bdata, 0, bdata.length);
414                        break;
415                    case DataBuffer.TYPE_SHORT:
416                        sdata = ((DataBufferShort)data).getData(bank);
417                        stream.writeShorts(sdata, 0, sdata.length);
418                        break;
419                    case DataBuffer.TYPE_USHORT:
420                        sdata = ((DataBufferUShort)data).getData(bank);
421                        stream.writeShorts(sdata, 0, sdata.length);
422                        break;
423                    case DataBuffer.TYPE_INT:
424                        idata = ((DataBufferInt)data).getData(bank);
425                        stream.writeInts(idata, 0, idata.length);
426                        break;
427                    case DataBuffer.TYPE_FLOAT:
428                        fdata = ((DataBufferFloat)data).getData(bank);
429                        stream.writeFloats(fdata, 0, fdata.length);
430                        break;
431                    case DataBuffer.TYPE_DOUBLE:
432                        ddata = ((DataBufferDouble)data).getData(bank);
433                        stream.writeDoubles(ddata, 0, ddata.length);
434                        break;
435                    }
436                }
437            } else {  // Single data bank
438                switch (type) {
439                    case DataBuffer.TYPE_BYTE:
440                        bdata = ((DataBufferByte)data).getData();
441                        break;
442                    case DataBuffer.TYPE_SHORT:
443                        sdata = ((DataBufferShort)data).getData();
444                        break;
445                    case DataBuffer.TYPE_USHORT:
446                        sdata = ((DataBufferUShort)data).getData();
447                        break;
448                    case DataBuffer.TYPE_INT:
449                        idata = ((DataBufferInt)data).getData();
450                        break;
451                    case DataBuffer.TYPE_FLOAT:
452                        fdata = ((DataBufferFloat)data).getData();
453                        break;
454                    case DataBuffer.TYPE_DOUBLE:
455                        ddata = ((DataBufferDouble)data).getData();
456                        break;
457                }
458
459                if (!noSubband &&
460                    bandStride >= raster.getWidth() *
461                                  raster.getHeight() * (numBands-1)) {
462
463                    for (int i = 0; i < numBands; i++) {
464                        int offset = bandOffsets[sourceBands[i]];
465                        switch (type) {
466                        case DataBuffer.TYPE_BYTE:
467                            stream.write(bdata, offset, bandSize);
468                            break;
469                        case DataBuffer.TYPE_SHORT:
470                        case DataBuffer.TYPE_USHORT:
471                            stream.writeShorts(sdata, offset, bandSize);
472                            break;
473                        case DataBuffer.TYPE_INT:
474                            stream.writeInts(idata, offset, bandSize);
475                            break;
476                        case DataBuffer.TYPE_FLOAT:
477                            stream.writeFloats(fdata, offset, bandSize);
478                            break;
479                        case DataBuffer.TYPE_DOUBLE:
480                            stream.writeDoubles(ddata, offset, bandSize);
481                            break;
482                        }
483                    }
484                } else {
485                    switch (type) {
486                        case DataBuffer.TYPE_BYTE:
487                            stream.write(bdata, 0, bdata.length);
488                            break;
489                        case DataBuffer.TYPE_SHORT:
490                        case DataBuffer.TYPE_USHORT:
491                            stream.writeShorts(sdata, 0, sdata.length);
492                            break;
493                        case DataBuffer.TYPE_INT:
494                            stream.writeInts(idata, 0, idata.length);
495                            break;
496                        case DataBuffer.TYPE_FLOAT:
497                            stream.writeFloats(fdata, 0, fdata.length);
498                            break;
499                        case DataBuffer.TYPE_DOUBLE:
500                            stream.writeDoubles(ddata, 0, ddata.length);
501                            break;
502                    }
503                }
504            }
505        } else if (sampleModel instanceof ComponentSampleModel) {
506            // The others, must be a ComponentSampleModel
507            switch (type) {
508                case DataBuffer.TYPE_BYTE:
509                    bdata = ((DataBufferByte)data).getData();
510                    break;
511                case DataBuffer.TYPE_SHORT:
512                    sdata = ((DataBufferShort)data).getData();
513                    break;
514                case DataBuffer.TYPE_USHORT:
515                    sdata = ((DataBufferUShort)data).getData();
516                    break;
517                case DataBuffer.TYPE_INT:
518                    idata = ((DataBufferInt)data).getData();
519                    break;
520                case DataBuffer.TYPE_FLOAT:
521                    fdata = ((DataBufferFloat)data).getData();
522                    break;
523                case DataBuffer.TYPE_DOUBLE:
524                    ddata = ((DataBufferDouble)data).getData();
525                    break;
526            }
527
528            ComponentSampleModel csm = (ComponentSampleModel)sampleModel;
529            int offset =
530                csm.getOffset(raster.getMinX()-raster.getSampleModelTranslateX(),
531                              raster.getMinY()-raster.getSampleModelTranslateY())
532                              - bandOffsets[0];
533
534            int srcSkip = pxlStride;
535            int copyLength = 1;
536            int innerStep = pxlStride;
537
538            int width = raster.getWidth();
539            int height = raster.getHeight();
540
541            int innerBound = width;
542            int outerBound = height;
543
544            if (srcSkip < lineStride) {
545                if (bandStride > pxlStride)
546                    copyLength = width;
547                srcSkip = lineStride;
548            } else {
549                if (bandStride > lineStride)
550                    copyLength = height;
551                innerStep = lineStride;
552                innerBound = height;
553                outerBound = width;
554            }
555
556            int writeLength = innerBound * numBands;
557            byte[] destBBuf = null;
558            short[] destSBuf = null;
559            int[] destIBuf = null;
560            float[] destFBuf = null;
561            double[] destDBuf = null;
562            Object srcBuf = null;
563            Object dstBuf = null;
564
565            switch (type) {
566            case DataBuffer.TYPE_BYTE:
567                srcBuf = bdata;
568                dstBuf = destBBuf = new byte[writeLength];
569                break;
570            case DataBuffer.TYPE_SHORT:
571            case DataBuffer.TYPE_USHORT:
572                srcBuf = sdata;
573                dstBuf = destSBuf = new short[writeLength];
574                break;
575            case DataBuffer.TYPE_INT:
576                srcBuf = idata;
577                dstBuf = destIBuf = new int[writeLength];
578                break;
579            case DataBuffer.TYPE_FLOAT:
580                srcBuf = fdata;
581                dstBuf = destFBuf = new float[writeLength];
582                break;
583            case DataBuffer.TYPE_DOUBLE:
584                srcBuf = ddata;
585                dstBuf = destDBuf = new double[writeLength];
586                break;
587            }
588
589            if (copyLength > 1) {
590                for (int i = 0; i < outerBound; i++) {
591                    for (int b = 0; b < numBands; b++) {
592                        int bandOffset = bandOffsets[b];
593
594                        System.arraycopy(srcBuf, offset + bandOffset,
595                                         dstBuf, b * innerBound, innerBound);
596                    }
597
598                    switch (type) {
599                    case DataBuffer.TYPE_BYTE:
600                        stream.write((byte[])dstBuf, 0, writeLength);
601                        break;
602                    case DataBuffer.TYPE_SHORT:
603                    case DataBuffer.TYPE_USHORT:
604                        stream.writeShorts((short[])dstBuf, 0, writeLength);
605                        break;
606                    case DataBuffer.TYPE_INT:
607                        stream.writeInts((int[])dstBuf, 0, writeLength);
608                        break;
609                    case DataBuffer.TYPE_FLOAT:
610                        stream.writeFloats((float[])dstBuf, 0, writeLength);
611                        break;
612                    case DataBuffer.TYPE_DOUBLE:
613                        stream.writeDoubles((double[])dstBuf, 0, writeLength);
614                        break;
615                    }
616                    offset += srcSkip;
617                }
618            } else {
619                switch (type) {
620                    case DataBuffer.TYPE_BYTE: {
621                        for (int i = 0; i < outerBound; i++) {
622                            for (int b = 0, k = 0; b < numBands; b++) {
623                                int bandOffset = bandOffsets[b];
624
625                                for (int j = 0, m = offset; j < innerBound;
626                                     j++, m += innerStep)
627                                    // copy one sample to the destination buffer
628                                    destBBuf[k++] = bdata[m + bandOffset];
629                            }
630
631                            stream.write(destBBuf, 0, writeLength);
632                            offset += srcSkip;
633                        }
634                    }
635                    break;
636                    case DataBuffer.TYPE_SHORT:
637                    case DataBuffer.TYPE_USHORT: {
638                        for (int i = 0; i < outerBound; i++) {
639                            for (int b = 0, k = 0; b < numBands; b++) {
640                                int bandOffset = bandOffsets[b];
641
642                                for (int j = 0, m = offset; j < innerBound;
643                                     j++, m += innerStep)
644                                    // copy one sample to the destination buffer
645                                    destSBuf[k++] = sdata[m + bandOffset];
646                            }
647
648                            stream.writeShorts(destSBuf, 0, writeLength);
649                            offset += srcSkip;
650                        }
651                    }
652                    break;
653                    case DataBuffer.TYPE_INT: {
654                        for (int i = 0; i < outerBound; i++) {
655                            for (int b = 0, k = 0; b < numBands; b++) {
656                                int bandOffset = bandOffsets[b];
657
658                                for (int j = 0, m = offset; j < innerBound;
659                                     j++, m += innerStep)
660                                    // copy one sample to the destination buffer
661                                    destIBuf[k++] = idata[m + bandOffset];
662                            }
663
664                            stream.writeInts(destIBuf, 0, writeLength);
665                            offset += srcSkip;
666                        }
667                    }
668                    break;
669                    case DataBuffer.TYPE_FLOAT: {
670                        for (int i = 0; i < outerBound; i++) {
671                            for (int b = 0, k = 0; b < numBands; b++) {
672                                int bandOffset = bandOffsets[b];
673
674                                for (int j = 0, m = offset; j < innerBound;
675                                     j++, m += innerStep)
676                                    // copy one sample to the destination buffer
677                                    destFBuf[k++] = fdata[m + bandOffset];
678                            }
679
680                            stream.writeFloats(destFBuf, 0, writeLength);
681                            offset += srcSkip;
682                        }
683                    }
684                    break;
685                    case DataBuffer.TYPE_DOUBLE: {
686                        for (int i = 0; i < outerBound; i++) {
687                            for (int b = 0, k = 0; b < numBands; b++) {
688                                int bandOffset = bandOffsets[b];
689
690                                for (int j = 0, m = offset; j < innerBound;
691                                     j++, m += innerStep)
692                                    // copy one sample to the destination buffer
693                                    destDBuf[k++] = ddata[m + bandOffset];
694                            }
695
696                            stream.writeDoubles(destDBuf, 0, writeLength);
697                            offset += srcSkip;
698                        }
699                    }
700                    break;
701                }
702            }
703        }
704    }

706    private Raster getTile(int tileX, int tileY) {
707        int sx = tileXOffset + tileX * tileWidth;
708        int sy = tileYOffset + tileY * tileHeight;
709        Rectangle bounds = new Rectangle(sx, sy, tileWidth, tileHeight);
710
711        if (writeRaster) {
712            bounds = bounds.intersection(destinationRegion);
713            if (noTransform) {
714                return inputRaster.createChild(bounds.x, bounds.y,
715                                           bounds.width, bounds.height,
716                                           bounds.x, bounds.y, sourceBands);
717            }
718
719            sx = bounds.x;
720            sy = bounds.y;
721
722            WritableRaster ras =
723                Raster.createWritableRaster(sampleModel, new Point(sx, sy));
724
725            int x = mapToSourceX(sx);
726            int y = mapToSourceY(sy);
727
728            int minY = inputRaster.getMinY();
729            int maxY = inputRaster.getMinY() + inputRaster.getHeight();
730
731            int cTileWidth = bounds.width;
732
733            int length = (cTileWidth - 1) * scaleX + 1;
734
735            for (int j = 0; j < bounds.height; j++, sy++, y += scaleY) {
736                if (y < minY || y >= maxY)
737                    continue;
738                Raster source =
739                    inputRaster.createChild(x, y, length, 1,
740                                            x, y, null);
741                int tempX = sx;
742                for (int i = 0, offset = x; i < cTileWidth;
743                    i++, tempX++, offset += scaleX) {
744                    for (int k = 0; k < numBands; k++) {
745                        int p = source.getSample(offset, y, sourceBands[k]);
746                        ras.setSample(tempX, sy, k, p);
747                    }
748                }
749            }
750
751            return ras;
752
753        } else {
754            if (noTransform) {
755                Raster ras = input.getTile(tileX, tileY);
756                if (destinationRegion.contains(bounds) && noSubband)
757                    return ras;
758                else {
759                    bounds = bounds.intersection(destinationRegion);
760                    return ras.createChild(bounds.x, bounds.y,
761                                           bounds.width, bounds.height,
762                                           bounds.x, bounds.y, sourceBands);
763                }
764            }
765
766            bounds = bounds.intersection(destinationRegion);
767            sx = bounds.x;
768            sy = bounds.y;
769
770            WritableRaster ras =
771                Raster.createWritableRaster(sampleModel, new Point(sx, sy));
772
773            int x = mapToSourceX(sx);
774            int y = mapToSourceY(sy);
775
776            int minY = input.getMinY();
777            int maxY = input.getMinY() + input.getHeight();
778
779            int cTileWidth = bounds.width;
780            int length = (cTileWidth -1) * scaleX + 1;
781
782            for (int j = 0; j < bounds.height; j++, sy++, y += scaleY) {
783                if (y < minY || y >= maxY)
784                    continue;
785
786                Raster source =
787                    input.getData(new Rectangle(x, y, length, 1));
788
789                int tempX = sx;
790                for (int i = 0, offset = x; i < cTileWidth;
791                    i++, tempX++, offset += scaleX) {
792                    for (int k = 0; k < numBands; k++) {
793                        int p = source.getSample(offset, y, sourceBands[k]);
794                        ras.setSample(tempX, sy, k, p);
795                    }
796                }
797            }
798            return ras;
799        }
800    }

802    private int mapToSourceX(int x) {
803        return x * scaleX + xOffset;
804    }
805
806    private int mapToSourceY(int y) {
807        return y * scaleY + yOffset;
808    }

810    private int getMinTileX() {
811        return ToTile(destinationRegion.x, tileXOffset, tileWidth);
812    }

814    private int getMaxTileX() {
815        return ToTile(destinationRegion.x + destinationRegion.width - 1,
816                      tileXOffset, tileWidth);

819    private int getMinTileY() {
820        return ToTile(destinationRegion.y, tileYOffset, tileHeight);
821    }

823    private int getMaxTileY() {
824        return ToTile(destinationRegion.y + destinationRegion.height - 1,
825                      tileYOffset, tileHeight);

828    private static int ToTile(int pos, int tileOffset, int tileSize) {
829        pos -= tileOffset;
830        if (pos < 0) {
831            pos += 1 - tileSize;         // force round to -infinity (ceiling)
832        }
833        return pos/tileSize;
834    }
