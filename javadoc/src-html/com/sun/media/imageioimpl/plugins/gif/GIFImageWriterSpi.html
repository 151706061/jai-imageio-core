046package com.sun.media.imageioimpl.plugins.gif;
047
048import java.awt.image.ColorModel;
049import java.awt.image.SampleModel;
050import java.util.Locale;
051import javax.imageio.ImageTypeSpecifier;
052import javax.imageio.ImageWriter;
053import javax.imageio.spi.ImageWriterSpi;
054import javax.imageio.spi.ServiceRegistry;
055import com.sun.media.imageioimpl.common.PaletteBuilder;
056import com.sun.media.imageioimpl.common.ImageUtil;
057import com.sun.media.imageioimpl.common.PackageUtil;
058
059public class GIFImageWriterSpi extends ImageWriterSpi {
060
061    private static final String vendorName = "Sun Microsystems, Inc.";
062
063    private static final String version = "1.0";
064
065    private static final String[] names = { "gif", "GIF" };
066
067    private static final String[] suffixes = { "gif" };
068
069    private static final String[] MIMETypes = { "image/gif" };
070
071    private static final String writerClassName =
072    "com.sun.media.imageioimpl.plugins.gif.GIFImageWriter";
073
074    private static final String[] readerSpiNames = {
075        "com.sun.imageio.plugins.gif.GIFImageReaderSpi"
076    };
077
078    private boolean registered = false;
079
080    public GIFImageWriterSpi() {
081        super(vendorName,
082              version,
083              names,
084              suffixes,
085              MIMETypes,
086              writerClassName,
087              STANDARD_OUTPUT_TYPE,
088              readerSpiNames,
089              true,
090              GIFWritableStreamMetadata.NATIVE_FORMAT_NAME,
091              "com.sun.media.imageioimpl.plugins.gif.GIFStreamMetadataFormat",
092              null, null,
093              true,
094              GIFWritableImageMetadata.NATIVE_FORMAT_NAME,
095              "com.sun.media.imageioimpl.plugins.gif.GIFStreamMetadataFormat",
096              null, null
097              );
098    }
099
100    public boolean canEncodeImage(ImageTypeSpecifier type) {
101        if (type == null) {
102            throw new IllegalArgumentException("type == null!");
103        }
104
105        SampleModel sm = type.getSampleModel();
106        ColorModel cm = type.getColorModel();
107
108        boolean canEncode = sm.getNumBands() == 1 &&
109            sm.getSampleSize(0) <= 8 &&
110            sm.getWidth() <= 65535 &&
111            sm.getHeight() <= 65535 &&
112            (cm == null || cm.getComponentSize()[0] <= 8);
113
114        if (canEncode) {
115            return true;
116        } else {
117            return PaletteBuilder.canCreatePalette(type);
118        }
119    }
120
121    public String getDescription(Locale locale) {
122        String desc = PackageUtil.getSpecificationTitle() + 
123            " GIF Image Writer";  
124        return desc;
125    }
126
127    public void onRegistration(ServiceRegistry registry,
128                               Class category) {
129        if (registered) {
130            return;
131        }
132        
133        registered = true;
134        
135        // By JDK 1.8, the GIFImageWriter will have been in JDK core for 
136        // atleast two FCS releases, so we can set JIIO's to lower priority
137        // With JDK 1.9, we can entirely de-register the JIIO one
138        ImageUtil.processOnRegistration(registry, category, "GIF", this,
139                                        9, 8); // JDK version 1.9, 1.8
140    }
141
142    public ImageWriter createWriterInstance(Object extension) {
143        return new GIFImageWriter(this);
144    }
145}
