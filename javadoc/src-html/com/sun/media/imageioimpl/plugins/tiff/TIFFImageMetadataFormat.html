054import javax.imageio.ImageTypeSpecifier;

056import com.sun.media.imageio.plugins.tiff.BaselineTIFFTagSet;
057import com.sun.media.imageio.plugins.tiff.TIFFTag;

059
060public class TIFFImageMetadataFormat extends TIFFMetadataFormat {
061
062    private static TIFFImageMetadataFormat theInstance = null;
063
064    static {
065    }
066
067    public boolean canNodeAppear(String elementName,
068                                 ImageTypeSpecifier imageType) {
069        return false;
070    }
071
072    private TIFFImageMetadataFormat() {
073        this.resourceBaseName =
074     "com.sun.media.imageioimpl.plugins.tiff.TIFFImageMetadataFormatResources";
075        this.rootName = TIFFImageMetadata.nativeMetadataFormatName;
076
077        TIFFElementInfo einfo;
078        TIFFAttrInfo ainfo;
079        String[] empty = new String[0];
080        String[] childNames;
081        String[] attrNames;
082
083        childNames = new String[] { "TIFFIFD" };
084        einfo = new TIFFElementInfo(childNames, empty, CHILD_POLICY_SEQUENCE);
085
086        elementInfoMap.put(TIFFImageMetadata.nativeMetadataFormatName,
087                           einfo);
088
089        childNames = new String[] { "TIFFField", "TIFFIFD" };
090        attrNames =
091            new String[] { "tagSets", "parentTagNumber", "parentTagName" };
092        einfo = new TIFFElementInfo(childNames, attrNames, CHILD_POLICY_SEQUENCE);
093        elementInfoMap.put("TIFFIFD", einfo);
094
095        ainfo = new TIFFAttrInfo();
096        ainfo.dataType = DATATYPE_STRING;
097        ainfo.isRequired = true;
098        attrInfoMap.put("TIFFIFD/tagSets", ainfo);
099
100        ainfo = new TIFFAttrInfo();
101        ainfo.dataType = DATATYPE_INTEGER;
102        ainfo.isRequired = false;
103        attrInfoMap.put("TIFFIFD/parentTagNumber", ainfo);
104
105        ainfo = new TIFFAttrInfo();
106        ainfo.dataType = DATATYPE_STRING;
107        ainfo.isRequired = false;
108        attrInfoMap.put("TIFFIFD/parentTagName", ainfo);
109
110        String[] types = {
111            "TIFFByte",
112            "TIFFAscii",
113            "TIFFShort",
114            "TIFFSShort",
115            "TIFFLong",
116            "TIFFSLong",
117            "TIFFRational",
118            "TIFFSRational",
119            "TIFFFloat",
120            "TIFFDouble",
121            "TIFFUndefined"
122        };
123
124        attrNames = new String[] { "value", "description" };
125        String[] attrNamesValueOnly = new String[] { "value" };
126        TIFFAttrInfo ainfoValue = new TIFFAttrInfo();
127        TIFFAttrInfo ainfoDescription = new TIFFAttrInfo();
128
129        for (int i = 0; i < types.length; i++) {
130            if (!types[i].equals("TIFFUndefined")) {
131                childNames = new String[1];
132                childNames[0] = types[i];
133                einfo =
134                    new TIFFElementInfo(childNames, empty, CHILD_POLICY_SEQUENCE);
135                elementInfoMap.put(types[i] + "s", einfo);
136            }
137
138            boolean hasDescription =
139                !types[i].equals("TIFFUndefined") &&
140                !types[i].equals("TIFFAscii") &&
141                !types[i].equals("TIFFRational") &&
142                !types[i].equals("TIFFSRational") &&
143                !types[i].equals("TIFFFloat") &&
144                !types[i].equals("TIFFDouble");
145
146            String[] anames = hasDescription ? attrNames : attrNamesValueOnly;
147            einfo = new TIFFElementInfo(empty, anames, CHILD_POLICY_EMPTY);
148            elementInfoMap.put(types[i], einfo);
149
150            attrInfoMap.put(types[i] + "/value", ainfoValue);
151            if (hasDescription) {
152                attrInfoMap.put(types[i] + "/description", ainfoDescription);
153            }
154        }
155            
156        childNames = new String[2*types.length - 1];
157        for (int i = 0; i < types.length; i++) {
158            childNames[2*i] = types[i];
159            if (!types[i].equals("TIFFUndefined")) {
160                childNames[2*i + 1] = types[i] + "s";
161            }
162        }
163        attrNames = new String[] { "number", "name" };
164        einfo = new TIFFElementInfo(childNames, attrNames, CHILD_POLICY_CHOICE);
165        elementInfoMap.put("TIFFField", einfo);
166
167        ainfo = new TIFFAttrInfo();
168        ainfo.isRequired = true;
169        attrInfoMap.put("TIFFField/number", ainfo);
170
171        ainfo = new TIFFAttrInfo();
172        attrInfoMap.put("TIFFField/name", ainfo);
173    }
174
175    public static synchronized IIOMetadataFormat getInstance() {
176        if (theInstance == null) {
177            theInstance = new TIFFImageMetadataFormat();
178        }
179        return theInstance;
180    }
181}
