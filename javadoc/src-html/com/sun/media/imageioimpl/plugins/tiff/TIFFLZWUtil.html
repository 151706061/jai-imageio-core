045package com.sun.media.imageioimpl.plugins.tiff;
046
047import java.awt.Rectangle;
048import java.io.IOException;
049import javax.imageio.IIOException;
050import javax.imageio.ImageReader;
051import com.sun.media.imageio.plugins.tiff.BaselineTIFFTagSet;
052import com.sun.media.imageio.plugins.tiff.TIFFTag;
053
054public class TIFFLZWUtil {
055
056    private static final boolean debug = false;
057
058    public TIFFLZWUtil() {
059    }
060
061    byte[] srcData;
062    int srcIndex;
063
064    byte[] dstData;
065    int dstIndex = 0;
066
067    byte stringTable[][];
068    int tableIndex, bitsToGet = 9;
069
070    int predictor, samplesPerPixel;
071    int nextData = 0;
072    int nextBits = 0;
073
074    private static final int andTable[] = {
075        511, 
076        1023,
077        2047,
078        4095
079    };
080
081    public byte[] decode(byte[] data, int predictor, int samplesPerPixel,
082                         int width, int height) throws IOException {
083        if (data[0] == (byte)0x00 && data[1] == (byte)0x01) {
084            throw new IIOException("TIFF 5.0-style LZW compression is not supported!");
085        }
086
087        this.srcData = data;
088        this.srcIndex = 0;
089        this.nextData = 0;
090        this.nextBits = 0;
091
092        this.dstData = new byte[8192];
093        this.dstIndex = 0;
094
095        initializeStringTable();
096
097        int code, oldCode = 0;
098        byte[] string;
099
100        while ((code = getNextCode()) != 257) {
101            if (code == 256) {
102                initializeStringTable();
103                code = getNextCode();
104                if (code == 257) {
105                    break;
106                }
107
108                writeString(stringTable[code]);
109                oldCode = code;
110            } else {
111                if (code < tableIndex) {
112                    string = stringTable[code];
113
114                    writeString(string);
115                    addStringToTable(stringTable[oldCode], string[0]); 
116                    oldCode = code;
117                } else {
118                    string = stringTable[oldCode];
119                    string = composeString(string, string[0]);
120                    writeString(string);
121                    addStringToTable(string);
122                    oldCode = code;
123                }
124            }
125        }
126
127        if (predictor == 2) {
128            
129            int count;
130            for (int j = 0; j < height; j++) {
131                
132                count = samplesPerPixel * (j * width + 1);
133                
134                for (int i = samplesPerPixel; i < width * samplesPerPixel; i++) {
135                    
136                    dstData[count] += dstData[count - samplesPerPixel];
137                    count++;
138                }
139            }
140        }
141
142        byte[] newDstData = new byte[dstIndex];
143        System.arraycopy(dstData, 0, newDstData, 0, dstIndex);
144        return newDstData;
145    }
146
147    /**
148     * Initialize the string table.
149     */
150    public void initializeStringTable() {
151        stringTable = new byte[4096][];
152        
153        for (int i = 0; i < 256; i++) {
154            stringTable[i] = new byte[1];
155            stringTable[i][0] = (byte)i;
156        }
157        
158        tableIndex = 258;
159        bitsToGet = 9;
160    }
161
162    private void ensureCapacity(int bytesToAdd) {
163        if (dstIndex + bytesToAdd > dstData.length) {
164            byte[] newDstData = new byte[Math.max((int)(dstData.length*1.2f),
165                                                  dstIndex + bytesToAdd)];
166            System.arraycopy(dstData, 0, newDstData, 0, dstData.length);
167            dstData = newDstData;
168        }
169    }
170
171    /**
172     * Write out the string just uncompressed.
173     */
174    public void writeString(byte string[]) {
175        ensureCapacity(string.length);
176        for (int i = 0; i < string.length; i++) {
177            dstData[dstIndex++] = string[i];
178        }
179    }
180    
181    /**
182     * Add a new string to the string table.
183     */
184    public void addStringToTable(byte oldString[], byte newString) {
185        int length = oldString.length;
186        byte string[] = new byte[length + 1];
187        System.arraycopy(oldString, 0, string, 0, length);
188        string[length] = newString;
189        
190        // Add this new String to the table
191        stringTable[tableIndex++] = string;
192        
193        if (tableIndex == 511) {
194            bitsToGet = 10;
195        } else if (tableIndex == 1023) {
196            bitsToGet = 11;
197        } else if (tableIndex == 2047) {
198            bitsToGet = 12;
199        } 
200    }
201
202    /**
203     * Add a new string to the string table.
204     */
205    public void addStringToTable(byte string[]) {
206        // Add this new String to the table
207        stringTable[tableIndex++] = string;
208        
209        if (tableIndex == 511) {
210            bitsToGet = 10;
211        } else if (tableIndex == 1023) {
212            bitsToGet = 11;
213        } else if (tableIndex == 2047) {
214            bitsToGet = 12;
215        } 
216    }
217
218    /**
219     * Append <code>newString</code> to the end of <code>oldString</code>.
220     */
221    public byte[] composeString(byte oldString[], byte newString) {
222        int length = oldString.length;
223        byte string[] = new byte[length + 1];
224        System.arraycopy(oldString, 0, string, 0, length);
225        string[length] = newString;
226
227        return string;
228    }
229
230    // Returns the next 9, 10, 11 or 12 bits
231    public int getNextCode() {
232        // Attempt to get the next code. The exception is caught to make
233        // this robust to cases wherein the EndOfInformation code has been
234        // omitted from a strip. Examples of such cases have been observed
235        // in practice.
236
237        try {
238            nextData = (nextData << 8) | (srcData[srcIndex++] & 0xff);
239            nextBits += 8;
240
241            if (nextBits < bitsToGet) {
242                nextData = (nextData << 8) | (srcData[srcIndex++] & 0xff);
243                nextBits += 8;
244            }
245
246            int code =
247                (nextData >> (nextBits - bitsToGet)) & andTable[bitsToGet - 9];
248            nextBits -= bitsToGet;
249
250            return code;
251        } catch (ArrayIndexOutOfBoundsException e) {
252            // Strip not terminated as expected: return EndOfInformation code.
253            return 257;
254        }
255    }
