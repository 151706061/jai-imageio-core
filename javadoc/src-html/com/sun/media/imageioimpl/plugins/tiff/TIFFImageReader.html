166    public void setInput(Object input,
167                         boolean seekForwardOnly,
168                         boolean ignoreMetadata) {
169        super.setInput(input, seekForwardOnly, ignoreMetadata);
170
171        // Clear all local values based on the previous stream contents.
172        resetLocal();
173
174        if (input != null) {
175            if (!(input instanceof ImageInputStream)) {
176                throw new IllegalArgumentException
177                    ("input not an ImageInputStream!"); 
178            }
179            this.stream = (ImageInputStream)input;
180        } else {
181            this.stream = null;
182        }
183    }

185    // Do not seek to the beginning of the stream so as to allow users to
186    // point us at an IFD within some other file format
187    private void readHeader() throws IIOException {
188        if (gotHeader) {
189            return;
190        }
191        if (stream == null) {
192            throw new IllegalStateException("Input not set!");
193        }
194
195        // Create an object to store the stream metadata
196        this.streamMetadata = new TIFFStreamMetadata();
197        
198        try {
199            int byteOrder = stream.readUnsignedShort();
200            if (byteOrder == 0x4d4d) {
201                streamMetadata.byteOrder = ByteOrder.BIG_ENDIAN;
202                stream.setByteOrder(ByteOrder.BIG_ENDIAN);
203            } else if (byteOrder == 0x4949) {
204                streamMetadata.byteOrder = ByteOrder.LITTLE_ENDIAN;
205                stream.setByteOrder(ByteOrder.LITTLE_ENDIAN);
206            } else {
207                processWarningOccurred(
208                           "Bad byte order in header, assuming little-endian");
209                streamMetadata.byteOrder = ByteOrder.LITTLE_ENDIAN;
210                stream.setByteOrder(ByteOrder.LITTLE_ENDIAN);
211            }
212            
213            int magic = stream.readUnsignedShort();
214            if (magic != 42) {
215                processWarningOccurred(
216                                     "Bad magic number in header, continuing");
217            }
218            
219            // Seek to start of first IFD
220            long offset = stream.readUnsignedInt();
221            imageStartPosition.add(new Long(offset));
222            stream.seek(offset);
223        } catch (IOException e) {
224            throw new IIOException("I/O error reading header!", e);
225        }
226
227        gotHeader = true;
228    }

230    private int locateImage(int imageIndex) throws IIOException {
231        readHeader();
232
233        try {
234            // Find closest known index
235            int index = Math.min(imageIndex, imageStartPosition.size() - 1);
236
237            // Seek to that position
238            Long l = (Long)imageStartPosition.get(index);
239            stream.seek(l.longValue());
240
241            // Skip IFDs until at desired index or last image found
242            while (index < imageIndex) {
243                int count = stream.readUnsignedShort();
244                stream.skipBytes(12*count);
245
246                long offset = stream.readUnsignedInt();
247                if (offset == 0) {
248                    return index;
249                }
250                
251                imageStartPosition.add(new Long(offset));
252                stream.seek(offset);
253                ++index;
254            }
255        } catch (IOException e) {
256            throw new IIOException("Couldn't seek!", e);
257        }
258
259        if (currIndex != imageIndex) {
260            imageMetadata = null;
261        }
262        currIndex = imageIndex;
263        return imageIndex;
264    }

266    public int getNumImages(boolean allowSearch) throws IOException {
267        if (stream == null) {
268            throw new IllegalStateException("Input not set!");
269        }
270        if (seekForwardOnly && allowSearch) {
271            throw new IllegalStateException
272                ("seekForwardOnly and allowSearch can't both be true!");
273        }
274
275        if (numImages > 0) {
276            return numImages;
277        }
278        if (allowSearch) {
279            this.numImages = locateImage(Integer.MAX_VALUE) + 1;
280        }
281        return numImages;
282    }

289    // Throw an IndexOutOfBoundsException if index < minIndex,
290    // and bump minIndex if required.
291    private void checkIndex(int imageIndex) {
292        if (imageIndex < minIndex) {
293            throw new IndexOutOfBoundsException("imageIndex < minIndex!");
294        }
295        if (seekForwardOnly) {
296            minIndex = imageIndex;
297        }
298    }

300    // Verify that imageIndex is in bounds, find the image IFD, read the
301    // image metadata, initialize instance variables from the metadata.
302    private void seekToImage(int imageIndex) throws IIOException {
303        checkIndex(imageIndex);
304
305        int index = locateImage(imageIndex);
306        if (index != imageIndex) {
307            throw new IndexOutOfBoundsException("imageIndex out of bounds!");
308        }
309
310        readMetadata();
311
312        initializeFromMetadata();
313    }

315    // Stream must be positioned at start of IFD for 'currIndex'
316    private void readMetadata() throws IIOException {
317        if (stream == null) {
318            throw new IllegalStateException("Input not set!");
319        }
320
321        if (imageMetadata != null) {
322            return;
323        }
324        try {
325            // Create an object to store the image metadata
326            List tagSets;
327            if (imageReadParam instanceof TIFFImageReadParam) {
328                tagSets =
329                    ((TIFFImageReadParam)imageReadParam).getAllowedTagSets();
330            } else {
331                tagSets = new ArrayList(1);
332                tagSets.add(BaselineTIFFTagSet.getInstance());
333            }
334
335            this.imageMetadata = new TIFFImageMetadata(tagSets);
336            imageMetadata.initializeFromStream(stream, ignoreMetadata);
337        } catch (IIOException iioe) {
338            throw iioe;
339        } catch (IOException ioe) {
340            throw new IIOException("I/O error reading image metadata!", ioe);
341        }
342    }

356    // Returns tile width if image is tiled, else image width
357    private int getTileOrStripWidth() {
358        TIFFField f =
359            imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_TILE_WIDTH);
360        return (f == null) ? getWidth() : f.getAsInt(0);
361    }

363    // Returns tile height if image is tiled, else strip height
364    private int getTileOrStripHeight() {
365        TIFFField f =
366            imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_TILE_LENGTH);
367        if (f != null) {
368            return f.getAsInt(0);
369        }
370
371        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_ROWS_PER_STRIP);
372        // Default for ROWS_PER_STRIP is 2^32 - 1, i.e., infinity
373        int h = (f == null) ? -1 : f.getAsInt(0);
374        return (h == -1) ? getHeight() : h;
375    }

377    private int getPlanarConfiguration() {
378        TIFFField f =
379        imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_PLANAR_CONFIGURATION);
380        if (f != null) {
381            int planarConfigurationValue = f.getAsInt(0);
382            if(planarConfigurationValue ==
383               BaselineTIFFTagSet.PLANAR_CONFIGURATION_PLANAR) {
384                // Some writers (e.g. Kofax standard Multi-Page TIFF
385                // Storage Filter v2.01.000; cf. bug 4929147) do not
386                // correctly set the value of this field. Attempt to
387                // ascertain whether the value is correctly Planar.
388                if(getCompression() ==
389                   BaselineTIFFTagSet.COMPRESSION_OLD_JPEG &&
390                   imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_JPEG_INTERCHANGE_FORMAT) !=
391                   null) {
392                    // JPEG interchange format cannot have
393                    // PlanarConfiguration value Chunky so reset.
394                    processWarningOccurred("PlanarConfiguration \"Planar\" value inconsistent with JPEGInterchangeFormat; resetting to \"Chunky\".");
395                    planarConfigurationValue =
396                        BaselineTIFFTagSet.PLANAR_CONFIGURATION_CHUNKY;
397                } else {
398                    TIFFField offsetField =
399                        imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_TILE_OFFSETS);
400                    if (offsetField == null) {
401                        // Tiles
402                        offsetField =
403                            imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_STRIP_OFFSETS);
404                        int tw = getTileOrStripWidth();
405                        int th = getTileOrStripHeight();
406                        int tAcross = (getWidth() + tw - 1)/tw;
407                        int tDown = (getHeight() + th - 1)/th;
408                        int tilesPerImage = tAcross*tDown;
409                        long[] offsetArray = offsetField.getAsLongs();
410                        if(offsetArray != null &&
411                           offsetArray.length == tilesPerImage) {
412                            // Length of offsets array is
413                            // TilesPerImage for Chunky and
414                            // SamplesPerPixel*TilesPerImage for Planar.
415                            processWarningOccurred("PlanarConfiguration \"Planar\" value inconsistent with TileOffsets field value count; resetting to \"Chunky\".");
416                            planarConfigurationValue =
417                                BaselineTIFFTagSet.PLANAR_CONFIGURATION_CHUNKY;
418                        }
419                    } else {
420                        // Strips
421                        int rowsPerStrip = getTileOrStripHeight();
422                        int stripsPerImage =
423                            (getHeight() + rowsPerStrip - 1)/rowsPerStrip;
424                        long[] offsetArray = offsetField.getAsLongs();
425                        if(offsetArray != null &&
426                           offsetArray.length == stripsPerImage) {
427                            // Length of offsets array is
428                            // StripsPerImage for Chunky and
429                            // SamplesPerPixel*StripsPerImage for Planar.
430                            processWarningOccurred("PlanarConfiguration \"Planar\" value inconsistent with StripOffsets field value count; resetting to \"Chunky\".");
431                            planarConfigurationValue =
432                                BaselineTIFFTagSet.PLANAR_CONFIGURATION_CHUNKY;
433                        }
434                    }
435                }
436            }
437            return planarConfigurationValue;
438        }
439
440        return BaselineTIFFTagSet.PLANAR_CONFIGURATION_CHUNKY;
441    }

443    private long getTileOrStripOffset(int tileIndex) throws IIOException {
444        TIFFField f =
445            imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_TILE_OFFSETS);
446        if (f == null) {
447            f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_STRIP_OFFSETS);
448        }
449        if (f == null) {
450            f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_JPEG_INTERCHANGE_FORMAT);
451        }
452
453        if(f == null) {
454            throw new IIOException
455                ("Missing required strip or tile offsets field.");
456        }
457
458        return f.getAsLong(tileIndex);
459    }

461    private long getTileOrStripByteCount(int tileIndex) throws IOException {
462        TIFFField f =
463           imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_TILE_BYTE_COUNTS);
464        if (f == null) {
465            f =
466          imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_STRIP_BYTE_COUNTS);
467        }
468        if (f == null) {
469            f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_JPEG_INTERCHANGE_FORMAT_LENGTH);
470        }
471
472        long tileOrStripByteCount;
473        if(f != null) {
474            tileOrStripByteCount = f.getAsLong(tileIndex);
475        } else {
476            processWarningOccurred("TIFF directory contains neither StripByteCounts nor TileByteCounts field: attempting to calculate from strip or tile width and height.");
477
478            // Initialize to number of bytes per strip or tile assuming
479            // no compression.
480            int bitsPerPixel = bitsPerSample[0];
481            for(int i = 1; i < samplesPerPixel; i++) {
482                bitsPerPixel += bitsPerSample[i];
483            }
484            int bytesPerRow = (getTileOrStripWidth()*bitsPerPixel + 7)/8;
485            tileOrStripByteCount = bytesPerRow*getTileOrStripHeight();
486
487            // Clamp to end of stream if possible.
488            long streamLength = stream.length();
489            if(streamLength != -1) {
490                tileOrStripByteCount =
491                    Math.min(tileOrStripByteCount,
492                             streamLength - getTileOrStripOffset(tileIndex));
493            } else {
494                processWarningOccurred("Stream length is unknown: cannot clamp estimated strip or tile byte count to EOF.");
495            }
496        }
497
498        return tileOrStripByteCount;
499    }

521    /**
522     * Initializes these instance variables from the image metadata:
523     * <pre>
524     * compression
525     * width
526     * height
527     * samplesPerPixel
528     * numBands
529     * colorMap
530     * photometricInterpretation
531     * sampleFormat
532     * bitsPerSample
533     * extraSamples
534     * tileOrStripWidth
535     * tileOrStripHeight
536     * </pre>
537     */
538    private void initializeFromMetadata() {
539        TIFFField f;
540
541        // Compression
542        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_COMPRESSION);
543        if (f == null) {
544            processWarningOccurred
545                ("Compression field is missing; assuming no compression");
546            compression = BaselineTIFFTagSet.COMPRESSION_NONE;
547        } else {
548            compression = f.getAsInt(0);
549        }
550
551        // Whether key dimensional information is absent.
552        boolean isMissingDimension = false;
553
554        // ImageWidth -> width
555        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_IMAGE_WIDTH);
556        if (f != null) {
557            this.width = f.getAsInt(0);
558        } else {
559            processWarningOccurred("ImageWidth field is missing.");
560            isMissingDimension = true;
561        }
562
563        // ImageLength -> height
564        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_IMAGE_LENGTH);
565        if (f != null) {
566            this.height = f.getAsInt(0);
567        } else {
568            processWarningOccurred("ImageLength field is missing.");
569            isMissingDimension = true;
570        }
571
572        // SamplesPerPixel
573        f =
574          imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_SAMPLES_PER_PIXEL);
575        if (f != null) {
576            samplesPerPixel = f.getAsInt(0);
577        } else {
578            samplesPerPixel = 1;
579            isMissingDimension = true;
580        }
581
582        // If any dimension is missing and there is a JPEG stream available
583        // get the information from it.
584        int defaultBitDepth = 1;
585        if(isMissingDimension &&
586           (f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_JPEG_INTERCHANGE_FORMAT)) != null) {
587            Iterator iter = ImageIO.getImageReadersByFormatName("JPEG");
588            if(iter != null && iter.hasNext()) {
589                ImageReader jreader = (ImageReader)iter.next();
590                try {
591                    stream.mark();
592                    stream.seek(f.getAsLong(0));
593                    jreader.setInput(stream);
594                    if(imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_IMAGE_WIDTH) == null) {
595                        this.width = jreader.getWidth(0);
596                    }
597                    if(imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_IMAGE_LENGTH) == null) {
598                        this.height = jreader.getHeight(0);
599                    }
600                    ImageTypeSpecifier imageType = jreader.getRawImageType(0);
601                    if(imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_SAMPLES_PER_PIXEL) == null) {
602                        this.samplesPerPixel =
603                            imageType.getSampleModel().getNumBands();
604                    }
605                    stream.reset();
606                    defaultBitDepth =
607                        imageType.getColorModel().getComponentSize(0);
608                } catch(IOException e) {
609                    // Ignore it and proceed: an error will occur later.
610                }
611                jreader.dispose();
612            }
613        }
614
615        if (samplesPerPixel < 1) {
616            processWarningOccurred("Samples per pixel < 1!");
617        }
618
619        // SamplesPerPixel -> numBands
620        numBands = samplesPerPixel;
621
622        // ColorMap
623        this.colorMap = null;
624        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_COLOR_MAP);
625        if (f != null) {
626            // Grab color map
627            colorMap = f.getAsChars();
628        }
629        
630        // PhotometricInterpretation
631        f =
632        imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_PHOTOMETRIC_INTERPRETATION);
633        if (f == null) {
634            if (compression == BaselineTIFFTagSet.COMPRESSION_CCITT_RLE ||
635                compression == BaselineTIFFTagSet.COMPRESSION_CCITT_T_4 ||
636                compression == BaselineTIFFTagSet.COMPRESSION_CCITT_T_6) {
637                processWarningOccurred
638                    ("PhotometricInterpretation field is missing; "+
639                     "assuming WhiteIsZero");
640                photometricInterpretation =
641                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_WHITE_IS_ZERO;
642            } else if(this.colorMap != null) {
643                photometricInterpretation =
644                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_PALETTE_COLOR;
645            } else if(samplesPerPixel == 3 || samplesPerPixel == 4) {
646                photometricInterpretation =
647                    BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_RGB;
648            } else {
649                processWarningOccurred
650                    ("PhotometricInterpretation field is missing; "+
651                     "assuming BlackIsZero");
652                photometricInterpretation =
653                   BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_BLACK_IS_ZERO;
654            }
655        } else {
656            photometricInterpretation = f.getAsInt(0);
657        }
658
659        // SampleFormat
660        boolean replicateFirst = false;
661        int first = -1;
662
663        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_SAMPLE_FORMAT);
664        sampleFormat = new int[samplesPerPixel];
665        replicateFirst = false;
666        if (f == null) {
667            replicateFirst = true;
668            first = BaselineTIFFTagSet.SAMPLE_FORMAT_UNDEFINED;
669        } else if (f.getCount() != samplesPerPixel) {
670            replicateFirst = true;
671            first = f.getAsInt(0);
672        }
673
674        for (int i = 0; i < samplesPerPixel; i++) {
675            sampleFormat[i] = replicateFirst ? first : f.getAsInt(i);
676            if (sampleFormat[i] !=
677                  BaselineTIFFTagSet.SAMPLE_FORMAT_UNSIGNED_INTEGER &&
678                sampleFormat[i] !=
679                  BaselineTIFFTagSet.SAMPLE_FORMAT_SIGNED_INTEGER &&
680                sampleFormat[i] !=
681                  BaselineTIFFTagSet.SAMPLE_FORMAT_FLOATING_POINT &&
682                sampleFormat[i] !=
683                  BaselineTIFFTagSet.SAMPLE_FORMAT_UNDEFINED) {
684                processWarningOccurred(
685          "Illegal value for SAMPLE_FORMAT, assuming SAMPLE_FORMAT_UNDEFINED");
686                sampleFormat[i] = BaselineTIFFTagSet.SAMPLE_FORMAT_UNDEFINED;
687            }
688        }
689
690        // BitsPerSample
691        f = imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_BITS_PER_SAMPLE);
692        this.bitsPerSample = new int[samplesPerPixel];

728    public Iterator getImageTypes(int imageIndex) throws IIOException {
729        List l; // List of ImageTypeSpecifiers
730
731        Integer imageIndexInteger = new Integer(imageIndex);
732        if(imageTypeMap.containsKey(imageIndexInteger)) {
733            // Return the cached ITS List.
734            l = (List)imageTypeMap.get(imageIndexInteger);
735        } else {
736            // Create a new ITS List.
737            l = new ArrayList(1);
738
739            // Create the ITS and cache if for later use so that this method
740            // always returns an Iterator containing the same ITS objects.
741            seekToImage(imageIndex);
742            ImageTypeSpecifier itsRaw = 
743                TIFFDecompressor.getRawImageTypeSpecifier
744                    (photometricInterpretation,
745                     compression,
746                     samplesPerPixel,
747                     bitsPerSample,
748                     sampleFormat,
749                     extraSamples,
750                     colorMap);
751
752            // Check for an ICCProfile field.
753            TIFFField iccProfileField =
754                imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_ICC_PROFILE);
755
756            // If an ICCProfile field is present change the ImageTypeSpecifier
757            // to use it if the data layout is component type.
758            if(iccProfileField != null &&
759               itsRaw.getColorModel() instanceof ComponentColorModel) {
760                // Create a ColorSpace from the profile.
761                byte[] iccProfileValue = iccProfileField.getAsBytes();
762                ICC_Profile iccProfile =
763                    ICC_Profile.getInstance(iccProfileValue);
764                ICC_ColorSpace iccColorSpace =
765                    new ICC_ColorSpace(iccProfile);
766
767                // Get the raw sample and color information.
768                ColorModel cmRaw = itsRaw.getColorModel();
769                ColorSpace csRaw = cmRaw.getColorSpace();
770                SampleModel smRaw = itsRaw.getSampleModel();
771
772                // Get the number of samples per pixel and the number
773                // of color components.
774                int numBands = smRaw.getNumBands();
775                int numComponents = iccColorSpace.getNumComponents();
776
777                // Replace the ColorModel with the ICC ColorModel if the
778                // numbers of samples and color components are amenable.
779                if(numBands == numComponents ||
780                   numBands == numComponents + 1) {
781                    // Set alpha flags.
782                    boolean hasAlpha = numComponents != numBands;
783                    boolean isAlphaPre =
784                        hasAlpha && cmRaw.isAlphaPremultiplied();
785
786                    // Create a ColorModel of the same class and with
787                    // the same transfer type.
788                    ColorModel iccColorModel =
789                        new ComponentColorModel(iccColorSpace,
790                                                cmRaw.getComponentSize(),
791                                                hasAlpha,
792                                                isAlphaPre,
793                                                cmRaw.getTransparency(),
794                                                cmRaw.getTransferType());
795
796                    // Prepend the ICC profile-based ITS to the List. The
797                    // ColorModel and SampleModel are guaranteed to be
798                    // compatible as the old and new ColorModels are both
799                    // ComponentColorModels with the same transfer type
800                    // and the same number of components.
801                    l.add(new ImageTypeSpecifier(iccColorModel, smRaw));
802
803                    // Append the raw ITS to the List if and only if its
804                    // ColorSpace has the same type and number of components
805                    // as the ICC ColorSpace.
806                    if(csRaw.getType() == iccColorSpace.getType() &&
807                       csRaw.getNumComponents() ==
808                       iccColorSpace.getNumComponents()) {
809                        l.add(itsRaw);
810                    }
811                } else { // ICCProfile not compatible with SampleModel.
812                    // Append the raw ITS to the List.
813                    l.add(itsRaw);
814                }
815            } else { // No ICCProfile field or raw ColorModel not component.
816                // Append the raw ITS to the List.
817                l.add(itsRaw);
818            }
819
820            // Cache the ITS List.
821            imageTypeMap.put(imageIndexInteger, l);
822        }
823
824        return l.iterator();
825    }

894        int h = getHeight(imageIndex);

900

902            throw new IllegalArgumentException
903                ("Tile indices are out of bounds!");
904        }
905
906        if (x + tw > w) {
907            tw = w - x;
908        }
909
910        if (y + th > h) {
911            th = h - y;
912        }
913
914        ImageReadParam param = getDefaultReadParam();
915        Rectangle tileRect = new Rectangle(x, y, tw, th);
916        param.setSourceRegion(tileRect);
917
918        return read(imageIndex, param);
919    }

943    private static int ifloor(int num, int den) {
944        if (num < 0) {
945            num -= den - 1;

958    private void prepareRead(int imageIndex, ImageReadParam param)
959        throws IOException {
960        if (stream == null) {
961            throw new IllegalStateException("Input not set!");
962        }
963
964        // A null ImageReadParam means we use the default
965        if (param == null) {
966            param = getDefaultReadParam();
967        }
968
969        this.imageReadParam = param;
970
971        seekToImage(imageIndex);
972
973        this.tileOrStripWidth = getTileOrStripWidth();
974        this.tileOrStripHeight = getTileOrStripHeight();
975        this.planarConfiguration = getPlanarConfiguration();
976
977        this.sourceBands = param.getSourceBands();
978        if (sourceBands == null) {
979            sourceBands = new int[numBands];
980            for (int i = 0; i < numBands; i++) {
981                sourceBands[i] = i;
982            }
983        }
984
985        // Initialize the destination image
986        Iterator imageTypes = getImageTypes(imageIndex);
987        ImageTypeSpecifier theImageType =
988            ImageUtil.getDestinationType(param, imageTypes);
989
990        int destNumBands = theImageType.getSampleModel().getNumBands();
991
992        this.destinationBands = param.getDestinationBands();
993        if (destinationBands == null) {
994            destinationBands = new int[destNumBands];
995            for (int i = 0; i < destNumBands; i++) {
996                destinationBands[i] = i;
997            }
998        }
999
1000        if (sourceBands.length != destinationBands.length) {
1001            throw new IllegalArgumentException(
1002                              "sourceBands.length != destinationBands.length");
1003        }
1004
1005        for (int i = 0; i < sourceBands.length; i++) {
1006            int sb = sourceBands[i];
1007            if (sb < 0 || sb >= numBands) {
1008                throw new IllegalArgumentException(
1009                                                  "Source band out of range!");
1010            }
1011            int db = destinationBands[i];
1012            if (db < 0 || db >= destNumBands) {
1013                throw new IllegalArgumentException(
1014                                             "Destination band out of range!");
1015            }
1016        }
1017    }

1027    private void decodeTile(int ti, int tj, int band) throws IOException {
1028        if(DEBUG) {
1029            System.out.println("decodeTile("+ti+","+tj+","+band+")");
1030        }
1031
1032        // Compute the region covered by the strip or tile
1033        Rectangle tileRect = new Rectangle(ti*tileOrStripWidth,
1034                                           tj*tileOrStripHeight,
1035                                           tileOrStripWidth,
1036                                           tileOrStripHeight);
1037
1038        // Clip against the image bounds if the image is not tiled. If it
1039        // is tiled, the tile may legally extend beyond the image bounds.
1040        if(!isImageTiled(currIndex)) {
1041            tileRect =
1042                tileRect.intersection(new Rectangle(0, 0, width, height));
1043        }
1044
1045        // Return if the intersection is empty.
1046        if(tileRect.width <= 0 || tileRect.height <= 0) {
1047            return;
1048        }
1049        
1050        int srcMinX = tileRect.x;
1051        int srcMinY = tileRect.y;
1052        int srcWidth = tileRect.width;
1053        int srcHeight = tileRect.height;
1054
1055        // Determine dest region that can be derived from the
1056        // source region
1057        
1058        dstMinX = iceil(srcMinX - sourceXOffset, srcXSubsampling);
1059        int dstMaxX = ifloor(srcMinX + srcWidth - 1 - sourceXOffset,
1060                         srcXSubsampling);
1061        
1062        dstMinY = iceil(srcMinY - sourceYOffset, srcYSubsampling);
1063        int dstMaxY = ifloor(srcMinY + srcHeight - 1 - sourceYOffset,
1064                             srcYSubsampling);
1065        
1066        dstWidth = dstMaxX - dstMinX + 1;
1067        dstHeight = dstMaxY - dstMinY + 1;
1068        
1069        dstMinX += dstXOffset;
1070        dstMinY += dstYOffset;
1071        
1072        // Clip against image bounds
1073        
1074        Rectangle dstRect = new Rectangle(dstMinX, dstMinY,
1075                                          dstWidth, dstHeight);
1076        dstRect =
1077            dstRect.intersection(theImage.getRaster().getBounds());
1078        
1079        dstMinX = dstRect.x;
1080        dstMinY = dstRect.y;
1081        dstWidth = dstRect.width;
1082        dstHeight = dstRect.height;
1083        
1084        if (dstWidth <= 0 || dstHeight <= 0) {
1085            return;
1086        }
1087        
1088        // Backwards map dest region to source to determine
1089        // active source region
1090        
1091        int activeSrcMinX = (dstMinX - dstXOffset)*srcXSubsampling +
1092            sourceXOffset;
1093        int sxmax = 
1094            (dstMinX + dstWidth - 1 - dstXOffset)*srcXSubsampling +
1095            sourceXOffset;
1096        int activeSrcWidth = sxmax - activeSrcMinX + 1;
1097        
1098        int activeSrcMinY = (dstMinY - dstYOffset)*srcYSubsampling +
1099            sourceYOffset;
1100        int symax =
1101            (dstMinY + dstHeight - 1 - dstYOffset)*srcYSubsampling +
1102            sourceYOffset;
1103        int activeSrcHeight = symax - activeSrcMinY + 1;
1104        
1105        decompressor.setSrcMinX(srcMinX);
1106        decompressor.setSrcMinY(srcMinY);
1107        decompressor.setSrcWidth(srcWidth);
1108        decompressor.setSrcHeight(srcHeight);
1109        
1110        decompressor.setDstMinX(dstMinX);
1111        decompressor.setDstMinY(dstMinY);
1112        decompressor.setDstWidth(dstWidth);
1113        decompressor.setDstHeight(dstHeight);
1114        
1115        decompressor.setActiveSrcMinX(activeSrcMinX);
1116        decompressor.setActiveSrcMinY(activeSrcMinY);
1117        decompressor.setActiveSrcWidth(activeSrcWidth);
1118        decompressor.setActiveSrcHeight(activeSrcHeight);
1119
1120        int tileIndex = tj*tilesAcross + ti;
1121
1122        if (planarConfiguration ==
1123            BaselineTIFFTagSet.PLANAR_CONFIGURATION_PLANAR) {
1124            tileIndex += band*tilesAcross*tilesDown;
1125        }
1126        
1127        long offset = getTileOrStripOffset(tileIndex);
1128        long byteCount = getTileOrStripByteCount(tileIndex);
1129
1130        //
1131        // Attempt to handle truncated streams, i.e., where reading the
1132        // compressed strip or tile would result in an EOFException. The
1133        // number of bytes to read is clamped to the number available
1134        // from the stream starting at the indicated position in the hope
1135        // that the decompressor will handle it.
1136        //
1137        long streamLength = stream.length();
1138        if(streamLength > 0 && offset + byteCount > streamLength) {
1139            processWarningOccurred("Attempting to process truncated stream.");
1140            if(Math.max(byteCount = streamLength - offset, 0) == 0) {
1141                processWarningOccurred("No bytes in strip/tile: skipping.");
1142                return;
1143            }
1144        }
1145
1146        decompressor.setStream(stream);
1147        decompressor.setOffset(offset);
1148        decompressor.setByteCount((int)byteCount);
1149        
1150        decompressor.beginDecoding();
1151
1152        stream.mark();
1153        decompressor.decode();
1154        stream.reset();
1155    }

1158        // Report image progress/update to listeners after each tile
1159        pixelsRead += dstWidth*dstHeight;
1160        processImageProgress(100.0f*pixelsRead/pixelsToRead);
1161        processImageUpdate(theImage,
1162                           dstMinX, dstMinY, dstWidth, dstHeight,
1163                           1, 1,
1164                           destinationBands);

1167    public BufferedImage read(int imageIndex, ImageReadParam param)
1168        throws IOException {
1169        prepareRead(imageIndex, param);
1170        this.theImage = getDestination(param,
1171                                       getImageTypes(imageIndex),
1172                                       width, height);
1173
1174        srcXSubsampling = imageReadParam.getSourceXSubsampling();
1175        srcYSubsampling = imageReadParam.getSourceYSubsampling();
1176
1177        Point p = imageReadParam.getDestinationOffset();
1178        dstXOffset = p.x;
1179        dstYOffset = p.y;
1180
1181        // This could probably be made more efficient...
1182        Rectangle srcRegion = new Rectangle(0, 0, 0, 0);
1183        Rectangle destRegion = new Rectangle(0, 0, 0, 0);
1184
1185        computeRegions(imageReadParam, width, height, theImage,
1186                       srcRegion, destRegion);
1187
1188        // Initial source pixel, taking source region and source
1189        // subsamplimg offsets into account
1190        sourceXOffset = srcRegion.x;
1191        sourceYOffset = srcRegion.y;
1192
1193        pixelsToRead = destRegion.width*destRegion.height;
1194        pixelsRead = 0;
1195
1196        processImageStarted(imageIndex);
1197        processImageProgress(0.0f);
1198
1199        tilesAcross = (width + tileOrStripWidth - 1)/tileOrStripWidth;
1200        tilesDown = (height + tileOrStripHeight - 1)/tileOrStripHeight;
1201
1202        int compression = getCompression();
1203
1204        // Attempt to get decompressor and color converted from the read param
1205        
1206        TIFFColorConverter colorConverter = null;
1207        if (imageReadParam instanceof TIFFImageReadParam) {
1208            TIFFImageReadParam tparam =
1209                (TIFFImageReadParam)imageReadParam;
1210            this.decompressor = tparam.getTIFFDecompressor();
1211            colorConverter = tparam.getColorConverter();
1212        }
1213
1214        // If we didn't find one, use a standard decompressor
1215        if (this.decompressor == null) {
1216            if (compression ==
1217                BaselineTIFFTagSet.COMPRESSION_NONE) {
1218                // Get the fillOrder field.
1219                TIFFField fillOrderField =
1220                    imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_FILL_ORDER);
1221
1222                // Set the decompressor based on the fill order.
1223                if(fillOrderField != null && fillOrderField.getAsInt(0) == 2) {
1224                    this.decompressor = new TIFFLSBDecompressor();
1225                } else {
1226                    this.decompressor = new TIFFNullDecompressor();
1227                }
1228            } else if (compression ==
1229                       BaselineTIFFTagSet.COMPRESSION_CCITT_T_6) {
1230
1231             
1232                // Fall back to the Java decompressor.
1233                if (this.decompressor == null) {
1234                    if(DEBUG) {
1235                        System.out.println("Using Java T.6 decompressor");
1236                    }
1237                    this.decompressor = new TIFFFaxDecompressor();
1238                }
1239            } else if (compression ==
1240                       BaselineTIFFTagSet.COMPRESSION_CCITT_T_4) {
1241
1242                   // Fall back to the Java decompressor.
1243                if (this.decompressor == null) {
1244                    if(DEBUG) {
1245                        System.out.println("Using Java T.4 decompressor");
1246                    }
1247                    this.decompressor = new TIFFFaxDecompressor();
1248                }
1249            } else if (compression ==
1250                       BaselineTIFFTagSet.COMPRESSION_CCITT_RLE) {
1251                this.decompressor = new TIFFFaxDecompressor();
1252            } else if (compression ==
1253                       BaselineTIFFTagSet.COMPRESSION_PACKBITS) {
1254                if(DEBUG) {
1255                    System.out.println("Using TIFFPackBitsDecompressor");
1256                }
1257                this.decompressor = new TIFFPackBitsDecompressor();
1258            } else if (compression ==
1259                       BaselineTIFFTagSet.COMPRESSION_LZW) {
1260                if(DEBUG) {
1261                    System.out.println("Using TIFFLZWDecompressor");
1262                }
1263                TIFFField predictorField =
1264                    imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_PREDICTOR);
1265                int predictor = ((predictorField == null) ?
1266                                 BaselineTIFFTagSet.PREDICTOR_NONE :
1267                                 predictorField.getAsInt(0));
1268                this.decompressor = new TIFFLZWDecompressor(predictor);
1269            } else if (compression ==
1270                       BaselineTIFFTagSet.COMPRESSION_JPEG) {
1271                this.decompressor = new TIFFJPEGDecompressor();
1272            } else if (compression ==
1273                       BaselineTIFFTagSet.COMPRESSION_ZLIB ||
1274                       compression ==
1275                       BaselineTIFFTagSet.COMPRESSION_DEFLATE) {
1276                TIFFField predictorField =
1277                    imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_PREDICTOR);
1278                int predictor = ((predictorField == null) ?
1279                                 BaselineTIFFTagSet.PREDICTOR_NONE :
1280                                 predictorField.getAsInt(0));
1281                this.decompressor = new TIFFDeflateDecompressor(predictor);
1282            } else if (compression ==
1283                       BaselineTIFFTagSet.COMPRESSION_OLD_JPEG) {
1284                TIFFField JPEGProcField =
1285                    imageMetadata.getTIFFField(BaselineTIFFTagSet.TAG_JPEG_PROC);
1286                if(JPEGProcField == null) {
1287                    processWarningOccurred
1288                        ("JPEGProc field missing; assuming baseline sequential JPEG process.");
1289                } else if(JPEGProcField.getAsInt(0) !=
1290                   BaselineTIFFTagSet.JPEG_PROC_BASELINE) {
1291                    throw new IIOException
1292                        ("Old-style JPEG supported for baseline sequential JPEG process only!");
1293                }
1294                this.decompressor = new TIFFOldJPEGDecompressor();
1295                //throw new IIOException("Old-style JPEG not supported!");
1296            } else {
1297                throw new IIOException
1298                    ("Unsupported compression type (tag number = "+
1299                     compression+")!");
1300            }
1301
1302            if (photometricInterpretation ==
1303                BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR &&
1304                compression != BaselineTIFFTagSet.COMPRESSION_JPEG &&
1305                compression != BaselineTIFFTagSet.COMPRESSION_OLD_JPEG) {
1306                boolean convertYCbCrToRGB =
1307                    theImage.getColorModel().getColorSpace().getType() ==
1308                    ColorSpace.TYPE_RGB;
1309                TIFFDecompressor wrappedDecompressor =
1310                    this.decompressor instanceof TIFFNullDecompressor ?
1311                    null : this.decompressor;
1312                this.decompressor =
1313                    new TIFFYCbCrDecompressor(wrappedDecompressor,
1314                                              convertYCbCrToRGB);
1315            }
1316        }
1317
1318        if(DEBUG) {
1319            System.out.println("\nDecompressor class = "+
1320                               decompressor.getClass().getName()+"\n");
1321        }
1322
1323        if (colorConverter == null) {
1324            if (photometricInterpretation ==
1325                BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_CIELAB &&
1326                theImage.getColorModel().getColorSpace().getType() ==
1327                ColorSpace.TYPE_RGB) {
1328                colorConverter = new TIFFCIELabColorConverter();
1329             } else if (photometricInterpretation ==
1330                        BaselineTIFFTagSet.PHOTOMETRIC_INTERPRETATION_Y_CB_CR &&
1331                        !(this.decompressor instanceof TIFFYCbCrDecompressor) &&
1332                        compression != BaselineTIFFTagSet.COMPRESSION_JPEG &&
1333                        compression != BaselineTIFFTagSet.COMPRESSION_OLD_JPEG) {
1334                 colorConverter = new TIFFYCbCrColorConverter(imageMetadata);
1335            }
1336        }
1337        
1338        decompressor.setReader(this);
1339        decompressor.setMetadata(imageMetadata);
1340        decompressor.setImage(theImage);
1341
1342        decompressor.setPhotometricInterpretation(photometricInterpretation);
1343        decompressor.setCompression(compression);
1344        decompressor.setSamplesPerPixel(samplesPerPixel);
1345        decompressor.setBitsPerSample(bitsPerSample);
1346        decompressor.setSampleFormat(sampleFormat);
1347        decompressor.setExtraSamples(extraSamples);
1348        decompressor.setColorMap(colorMap);
1349

1364        int minTileX =
1365            TIFFImageWriter.XToTileX(srcRegion.x, 0, tileOrStripWidth);

1367            TIFFImageWriter.YToTileY(srcRegion.y, 0, tileOrStripHeight);

1369            TIFFImageWriter.XToTileX(srcRegion.x + srcRegion.width - 1,
1370                                     0, tileOrStripWidth);
1371        int maxTileY =
1372            TIFFImageWriter.YToTileY(srcRegion.y + srcRegion.height - 1,
1373                                     0, tileOrStripHeight);
1374
1375        boolean isAbortRequested = false;
1376        if (planarConfiguration ==
1377            BaselineTIFFTagSet.PLANAR_CONFIGURATION_PLANAR) {
1378            
1379            decompressor.setPlanar(true);
1380            
1381            int[] sb = new int[1];
1382            int[] db = new int[1];
1383            for (int tj = minTileY; tj <= maxTileY; tj++) {
1384                for (int ti = minTileX; ti <= maxTileX; ti++) {
1385                    for (int band = 0; band < numBands; band++) {
1386                        sb[0] = sourceBands[band];
1387                        decompressor.setSourceBands(sb);
1388                        db[0] = destinationBands[band];
1389                        decompressor.setDestinationBands(db);
1390                        //XXX decompressor.beginDecoding();
1391
1392                        // The method abortRequested() is synchronized
1393                        // so check it only once per loop just before
1394                        // doing any actual decoding.
1395                        if(abortRequested()) {
1396                            isAbortRequested = true;
1397                            break;
1398                        }
1399
1400                        decodeTile(ti, tj, band);
1401                    }
1402
1403                    if(isAbortRequested) break;
1404
1405                    reportProgress();
1406                }
1407
1408                if(isAbortRequested) break;
1409            }
1410        } else {
1411            //XXX decompressor.beginDecoding();
1412
1413            for (int tj = minTileY; tj <= maxTileY; tj++) {
1414                for (int ti = minTileX; ti <= maxTileX; ti++) {
1415                    // The method abortRequested() is synchronized
1416                    // so check it only once per loop just before
1417                    // doing any actual decoding.
1418                    if(abortRequested()) {
1419                        isAbortRequested = true;
1420                        break;
1421                    }
1422
1423                    decodeTile(ti, tj, -1);
1424
1425                    reportProgress();
1426                }
1427
1428                if(isAbortRequested) break;
1429            }
1430        }
1431
1432        if (isAbortRequested) {
1433            processReadAborted();
1434        } else {
1435            processImageComplete();
1436        }
1437
1438        return theImage;

1473    protected static BufferedImage getDestination(ImageReadParam param,
1474                                                  Iterator imageTypes,
1475                                                  int width, int height)
1476            throws IIOException {
1477        if (imageTypes == null || !imageTypes.hasNext()) {
1478            throw new IllegalArgumentException("imageTypes null or empty!");
1479        }        
1480        
1481        BufferedImage dest = null;
1482        ImageTypeSpecifier imageType = null;
1483
1484        // If param is non-null, use it
1485        if (param != null) {
1486            // Try to get the image itself
1487            dest = param.getDestination();
1488            if (dest != null) {
1489                return dest;
1490            }
1491        
1492            // No image, get the image type
1493            imageType = param.getDestinationType();
1494        }
1495
1496        // No info from param, use fallback image type
1497        if (imageType == null) {
1498            Object o = imageTypes.next();
1499            if (!(o instanceof ImageTypeSpecifier)) {
1500                throw new IllegalArgumentException
1501                    ("Non-ImageTypeSpecifier retrieved from imageTypes!");
1502            }
1503            imageType = (ImageTypeSpecifier)o;
1504        } else {
1505            boolean foundIt = false;
1506            while (imageTypes.hasNext()) {
1507                ImageTypeSpecifier type =
1508                    (ImageTypeSpecifier)imageTypes.next();
1509                if (type.equals(imageType)) {
1510                    foundIt = true;
1511                    break;
1512                }
1513            }
1514
1515            if (!foundIt) {
1516                throw new IIOException
1517                    ("Destination type from ImageReadParam does not match!");
1518            }
1519        }
1520
1521        Rectangle srcRegion = new Rectangle(0,0,0,0);
1522        Rectangle destRegion = new Rectangle(0,0,0,0);
1523        computeRegions(param,
1524                       width,
1525                       height,
1526                       null,
1527                       srcRegion,
1528                       destRegion);
1529        
1530        int destWidth = destRegion.x + destRegion.width;
1531        int destHeight = destRegion.y + destRegion.height;
1532        // Create a new image based on the type specifier
1533        
1534        if ((long)destWidth*destHeight > Integer.MAX_VALUE) {
1535            throw new IllegalArgumentException
1536                ("width*height > Integer.MAX_VALUE!");
1537        }
1538        
1539        return imageType.createBufferedImage(destWidth, destHeight);
1540    }
