045package com.sun.media.imageioimpl.plugins.tiff;
046
047import java.awt.Rectangle;
048import java.io.IOException;
049import javax.imageio.ImageReader;
050import com.sun.media.imageio.plugins.tiff.BaselineTIFFTagSet;
051import com.sun.media.imageio.plugins.tiff.TIFFTag;
052
053public class TIFFPackBitsUtil {
054
055    private static final boolean debug = false;
056
057    byte[] dstData = new byte[8192];
058    int dstIndex = 0;
059
060    public TIFFPackBitsUtil() {
061    }
062
063    private void ensureCapacity(int bytesToAdd) {
064        if (dstIndex + bytesToAdd > dstData.length) {
065            byte[] newDstData = new byte[Math.max((int)(dstData.length*1.2f),
066                                                  dstIndex + bytesToAdd)];
067            System.arraycopy(dstData, 0, newDstData, 0, dstData.length);
068            dstData = newDstData;
069        }
070    }
071
072    public byte[] decode(byte[] srcData) throws IOException {
073        int inIndex = 0;
074        while (inIndex < srcData.length) {
075            byte b = srcData[inIndex++];
076            
077            if (b >= 0 && b <= 127) { // second test not needed?
078                // Literal run packet
079                
080                ensureCapacity(b + 1);
081                for (int i = 0; i < b + 1; i++) {
082                    dstData[dstIndex++] = srcData[inIndex++];
083                }
084            } else if (b <= -1 && b >= -127) {
085                // 2-byte encoded run packet
086                byte repeat = srcData[inIndex++];
087                ensureCapacity(-b + 1);
088                for (int i = 0; i < (-b + 1); i++) {
089                    dstData[dstIndex++] = repeat;
090                }
091            } else {
092                // No-op packet, do nothing
093                ++inIndex;
094            }
095        }
096        
097        byte[] newDstData = new byte[dstIndex];
098        System.arraycopy(dstData, 0, newDstData, 0, dstIndex);
099        return newDstData;
100    }
101}
