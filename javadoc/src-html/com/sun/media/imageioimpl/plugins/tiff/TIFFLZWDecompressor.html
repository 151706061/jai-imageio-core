045package com.sun.media.imageioimpl.plugins.tiff;

047import java.awt.Rectangle;
048import java.io.IOException;
049import javax.imageio.IIOException;
050import javax.imageio.ImageReader;
051import com.sun.media.imageio.plugins.tiff.BaselineTIFFTagSet;
052import com.sun.media.imageio.plugins.tiff.TIFFDecompressor;

054
055public class TIFFLZWDecompressor extends TIFFDecompressor {
056
057    private static final boolean DEBUG = false;
058
059    private static final int andTable[] = {
060        511, 
061        1023,
062        2047,
063        4095
064    };
065
066    int predictor;
067
068    byte[] srcData;
069    byte[] dstData;
070
071    int srcIndex;
072    int dstIndex;
073
074    byte stringTable[][];
075    int tableIndex, bitsToGet = 9;
076
077    int nextData = 0;
078    int nextBits = 0;
079
080    public TIFFLZWDecompressor(int predictor) throws IIOException {
081        super();
082
083        if (predictor != BaselineTIFFTagSet.PREDICTOR_NONE && 
084            predictor != 
085            BaselineTIFFTagSet.PREDICTOR_HORIZONTAL_DIFFERENCING) {
086            throw new IIOException("Illegal value for Predictor in " +
087                                   "TIFF file");
088        }
089
090        if(DEBUG) {
091            System.out.println("Using horizontal differencing predictor");
092        }
093
094        this.predictor = predictor;
095    }
096
097    public void decodeRaw(byte[] b,
098                          int dstOffset,
099                          int bitsPerPixel,
100                          int scanlineStride) throws IOException {
101
102        // Check bitsPerSample.
103        if (predictor == 
104            BaselineTIFFTagSet.PREDICTOR_HORIZONTAL_DIFFERENCING) {
105            int len = bitsPerSample.length;
106            for(int i = 0; i < len; i++) {
107                if(bitsPerSample[i] != 8) {
108                    throw new IIOException
109                        (bitsPerSample[i] + "-bit samples "+
110                         "are not supported for Horizontal "+
111                         "differencing Predictor");
112                }
113            }
114        }
115
116        stream.seek(offset);
117
118        byte[] sdata = new byte[byteCount];
119        stream.readFully(sdata);
120
121        int bytesPerRow = (srcWidth*bitsPerPixel + 7)/8;
122        byte[] buf;
123        int bufOffset;
124        if(bytesPerRow == scanlineStride) {
125            buf = b;
126            bufOffset = dstOffset;
127        } else {
128            buf = new byte[bytesPerRow*srcHeight];
129            bufOffset = 0;
130        }
131
132        int numBytesDecoded = decode(sdata, 0, buf, bufOffset);
133
134        if(bytesPerRow != scanlineStride) {
135            if(DEBUG) {
136                System.out.println("bytesPerRow != scanlineStride");
137            }
138            int off = 0;
139            for (int y = 0; y < srcHeight; y++) {
140                System.arraycopy(buf, off, b, dstOffset, bytesPerRow);
141                off += bytesPerRow;
142                dstOffset += scanlineStride;
143            }
144        }
145    }
146
147    public int decode(byte[] sdata, int srcOffset,
148                      byte[] ddata, int dstOffset)
149        throws IOException {
150        if (sdata[0] == (byte)0x00 && sdata[1] == (byte)0x01) {
151            throw new IIOException
152                ("TIFF 5.0-style LZW compression is not supported!");
153        }
154
155        this.srcData = sdata;
156        this.dstData = ddata;
157
158        this.srcIndex = srcOffset;
159        this.dstIndex = dstOffset;
160
161        this.nextData = 0;
162        this.nextBits = 0;
163
164        initializeStringTable();
165
166        int code, oldCode = 0;
167        byte[] string;
168
169        while ((code = getNextCode()) != 257) {
170            if (code == 256) {
171                initializeStringTable();
172                code = getNextCode();
173                if (code == 257) {
174                    break;
175                }
176
177                writeString(stringTable[code]);
178                oldCode = code;
179            } else {
180                if (code < tableIndex) {
181                    string = stringTable[code];
182
183                    writeString(string);
184                    addStringToTable(stringTable[oldCode], string[0]); 
185                    oldCode = code;
186                } else {
187                    string = stringTable[oldCode];
188                    string = composeString(string, string[0]);
189                    writeString(string);
190                    addStringToTable(string);
191                    oldCode = code;
192                }
193            }
194        }
195
196        if (predictor ==
197            BaselineTIFFTagSet.PREDICTOR_HORIZONTAL_DIFFERENCING) {
198
199            for (int j = 0; j < srcHeight; j++) {
200                
201                int count = dstOffset + samplesPerPixel * (j * srcWidth + 1);
202                
203                for (int i = samplesPerPixel; i < srcWidth * samplesPerPixel; i++) {
204                    
205                    dstData[count] += dstData[count - samplesPerPixel];
206                    count++;
207                }
208            }
209        }
210
211        return dstIndex - dstOffset;
212    }
213
214    /**
215     * Initialize the string table.
216     */
217    public void initializeStringTable() {
218        stringTable = new byte[4096][];
219        
220        for (int i = 0; i < 256; i++) {
221            stringTable[i] = new byte[1];
222            stringTable[i][0] = (byte)i;
223        }
224        
225        tableIndex = 258;
226        bitsToGet = 9;
227    }
228
229    /**
230     * Write out the string just uncompressed.
231     */
232    public void writeString(byte string[]) {
233        if(dstIndex < dstData.length) {
234            int maxIndex = Math.min(string.length,
235                                    dstData.length - dstIndex);
236
237            for (int i=0; i < maxIndex; i++) {
238                dstData[dstIndex++] = string[i];
239            }
240        }
241    }
242    
243    /**
244     * Add a new string to the string table.
245     */
246    public void addStringToTable(byte oldString[], byte newString) {
247        int length = oldString.length;
248        byte string[] = new byte[length + 1];
249        System.arraycopy(oldString, 0, string, 0, length);
250        string[length] = newString;
251        
252        // Add this new String to the table
253        stringTable[tableIndex++] = string;
254        
255        if (tableIndex == 511) {
256            bitsToGet = 10;
257        } else if (tableIndex == 1023) {
258            bitsToGet = 11;
259        } else if (tableIndex == 2047) {
260            bitsToGet = 12;
261        } 
262    }
263
264    /**
265     * Add a new string to the string table.
266     */
267    public void addStringToTable(byte string[]) {
268        // Add this new String to the table
269        stringTable[tableIndex++] = string;
270        
271        if (tableIndex == 511) {
272            bitsToGet = 10;
273        } else if (tableIndex == 1023) {
274            bitsToGet = 11;
275        } else if (tableIndex == 2047) {
276            bitsToGet = 12;
277        } 
278    }
279
280    /**
281     * Append <code>newString</code> to the end of <code>oldString</code>.
282     */
283    public byte[] composeString(byte oldString[], byte newString) {
284        int length = oldString.length;
285        byte string[] = new byte[length + 1];
286        System.arraycopy(oldString, 0, string, 0, length);
287        string[length] = newString;
288
289        return string;
290    }
291
292    // Returns the next 9, 10, 11 or 12 bits
293    public int getNextCode() {
294        // Attempt to get the next code. The exception is caught to make
295        // this robust to cases wherein the EndOfInformation code has been
296        // omitted from a strip. Examples of such cases have been observed
297        // in practice.
298
299        try {
300            nextData = (nextData << 8) | (srcData[srcIndex++] & 0xff);
301            nextBits += 8;
302
303            if (nextBits < bitsToGet) {
304                nextData = (nextData << 8) | (srcData[srcIndex++] & 0xff);
305                nextBits += 8;
306            }
307
308            int code =
309                (nextData >> (nextBits - bitsToGet)) & andTable[bitsToGet - 9];
310            nextBits -= bitsToGet;
311
312            return code;
313        } catch (ArrayIndexOutOfBoundsException e) {
314            // Strip not terminated as expected: return EndOfInformation code.
315            return 257;
316        }
317    }
