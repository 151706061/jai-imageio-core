080
081    public TIFFRenderedImage(TIFFImageReader reader,
082                             int imageIndex,
083                             ImageReadParam readParam,
084                             int width, int height) throws IOException {
085        this.reader = reader;
086        this.imageIndex = imageIndex;
087        this.tileParam = cloneImageReadParam(readParam, false);
088
089        this.subsampleX = tileParam.getSourceXSubsampling();
090        this.subsampleY = tileParam.getSourceYSubsampling();
091
092        this.isSubsampling = this.subsampleX != 1 || this.subsampleY != 1;
093
094        this.width = width/subsampleX;
095        this.height = height/subsampleY;
096
097        // If subsampling is being used, we may not match the
098        // true tile grid exactly, but everything should still work
099        this.tileWidth = reader.getTileWidth(imageIndex)/subsampleX;
100        this.tileHeight = reader.getTileHeight(imageIndex)/subsampleY;
101        
102        Iterator iter = reader.getImageTypes(imageIndex);
103        this.its = (ImageTypeSpecifier)iter.next();
104        tileParam.setDestinationType(its);
105    }
106
107    /**
108     * Creates a copy of <code>param</code>. The source subsampling and
109     * and bands settings and the destination bands and offset settings
110     * are copied. If <code>param</code> is a <code>TIFFImageReadParam</code>
111     * then the <code>TIFFDecompressor</code> and
112     * <code>TIFFColorConverter</code> settings are also copied; otherwise
113     * they are explicitly set to <code>null</code>.
114     *
115     * @param param the parameters to be copied.
116     * @param copyTagSets whether the <code>TIFFTagSet</code> settings
117     * should be copied if set.
118     * @return copied parameters.
119     */
120    private ImageReadParam cloneImageReadParam(ImageReadParam param,
121                                               boolean copyTagSets) {
122        // Create a new TIFFImageReadParam.
123        TIFFImageReadParam newParam = new TIFFImageReadParam();
124
125        // Copy the basic settings.
126        newParam.setSourceSubsampling(param.getSourceXSubsampling(),
127                                      param.getSourceYSubsampling(),
128                                      param.getSubsamplingXOffset(),
129                                      param.getSubsamplingYOffset());
130        newParam.setSourceBands(param.getSourceBands());
131        newParam.setDestinationBands(param.getDestinationBands());
132        newParam.setDestinationOffset(param.getDestinationOffset());
133
134        // Set the decompressor and color converter.
135        if(param instanceof TIFFImageReadParam) {
136            // Copy the settings from the input parameter.
137            TIFFImageReadParam tparam = (TIFFImageReadParam)param;
138            newParam.setTIFFDecompressor(tparam.getTIFFDecompressor());
139            newParam.setColorConverter(tparam.getColorConverter());
140
141            if(copyTagSets) {
142                List tagSets = tparam.getAllowedTagSets();
143                if(tagSets != null) {
144                    Iterator tagSetIter = tagSets.iterator();
145                    if(tagSetIter != null) {
146                        while(tagSetIter.hasNext()) {
147                            TIFFTagSet tagSet = (TIFFTagSet)tagSetIter.next();
148                            newParam.addAllowedTagSet(tagSet);
149                        }
150                    }
151                }
152            }
153        } else {
154            // Set the decompressor and color converter to null.
155            newParam.setTIFFDecompressor(null);
156            newParam.setColorConverter(null);
157        }
158
159        return newParam;
160    }

230    public Raster getTile(int tileX, int tileY) {
231        Rectangle tileRect = new Rectangle(tileX*tileWidth,
232                                           tileY*tileHeight,
233                                           tileWidth,
234                                           tileHeight);
235        return getData(tileRect);
236    }

238    public Raster getData() {
239        return read(new Rectangle(0, 0, getWidth(), getHeight()));
240    }

242    public Raster getData(Rectangle rect) {
243        return read(rect);
244    }
245
246    // This method needs to be synchronized as it updates the instance
247    // variable 'tileParam'.
248    public synchronized WritableRaster read(Rectangle rect) {
249        // XXX Does this need to consider the subsampling offsets or is
250        // that handled implicitly by the reader?
251        tileParam.setSourceRegion(isSubsampling ?
252                                  new Rectangle(subsampleX*rect.x,
253                                                subsampleY*rect.y,
254                                                subsampleX*rect.width,
255                                                subsampleY*rect.height) :
256                                  rect);
257
258        try {
259            BufferedImage bi = reader.read(imageIndex, tileParam);
260            WritableRaster ras = bi.getRaster();
261            return ras.createWritableChild(0, 0,
262                                           ras.getWidth(), ras.getHeight(),
263                                           rect.x, rect.y,
264                                           null);
265        } catch (IOException e) {
266            throw new RuntimeException(e);
267        }
268    }
269
270    public WritableRaster copyData(WritableRaster raster) {
271        if (raster == null) {
272            return read(new Rectangle(0, 0, getWidth(), getHeight()));
273        } else {
274            Raster src = read(raster.getBounds());
275            raster.setRect(src);
276            return raster;
277        }
278    }
