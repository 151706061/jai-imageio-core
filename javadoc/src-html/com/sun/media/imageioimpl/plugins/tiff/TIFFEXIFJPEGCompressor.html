065/**
066 * A <code>TIFFCompressor</code> for the JPEG variant of EXIF.
067 */
068public class TIFFEXIFJPEGCompressor extends TIFFBaseJPEGCompressor {
069    public TIFFEXIFJPEGCompressor(ImageWriteParam param) {
070        super(TIFFImageWriter.EXIF_JPEG_COMPRESSION_TYPE,
071              BaselineTIFFTagSet.COMPRESSION_OLD_JPEG,
072              false,
073              param);
074    }
075
076    public void setMetadata(IIOMetadata metadata) {
077        // Set the metadata.
078        super.setMetadata(metadata);
079
080        // Initialize the JPEG writer and writeparam.
081        initJPEGWriter(false, // No stream metadata (not writing abbreviated)
082                       true); // Yes image metadata (remove APPn markers)
083    }
084}
