045package com.sun.media.imageioimpl.plugins.tiff;
046
047import java.io.IOException;
048import com.sun.media.imageio.plugins.tiff.TIFFDecompressor;
049
050public class TIFFNullDecompressor extends TIFFDecompressor {
051
052    private static final boolean DEBUG = false; // XXX false for release!
053
054    /**
055     * Whether to read the active source region only.
056     */
057    private boolean isReadActiveOnly = false;
058
059    /** The original value of <code>srcMinX</code>. */
060    private int originalSrcMinX;
061
062    /** The original value of <code>srcMinY</code>. */
063    private int originalSrcMinY;
064
065    /** The original value of <code>srcWidth</code>. */
066    private int originalSrcWidth;
067
068    /** The original value of <code>srcHeight</code>. */
069    private int originalSrcHeight;
070
071    public TIFFNullDecompressor() {}
072
073    //
074    // This approach to reading the active reading is a hack of sorts
075    // as the original values of the entire source region are stored,
076    // overwritten, and then restored. It would probably be better to
077    // revise TIFFDecompressor such that this were not necessary, i.e.,
078    // change beginDecoding() and decode() to use the active region values
079    // when random access is easy and the entire region values otherwise.
080    //
081    public void beginDecoding() {
082        // Determine number of bits per pixel.
083        int bitsPerPixel = 0;
084        for(int i = 0; i < bitsPerSample.length; i++) {
085            bitsPerPixel += bitsPerSample[i];
086        }
087
088        // Can read active region only if row starts on a byte boundary.
089        if((activeSrcMinX != srcMinX || activeSrcMinY != srcMinY ||
090            activeSrcWidth != srcWidth || activeSrcHeight != srcHeight) &&
091           ((activeSrcMinX - srcMinX)*bitsPerPixel) % 8 == 0) {
092            // Set flag.
093            isReadActiveOnly = true;
094
095            // Cache original region.
096            originalSrcMinX = srcMinX;
097            originalSrcMinY = srcMinY;
098            originalSrcWidth = srcWidth;
099            originalSrcHeight = srcHeight;
100
101            // Replace region with active region.
102            srcMinX = activeSrcMinX;
103            srcMinY = activeSrcMinY;
104            srcWidth = activeSrcWidth;
105            srcHeight = activeSrcHeight;
106        } else {
107            // Clear flag.
108            isReadActiveOnly = false;
109        }
110
111        if(DEBUG) {
112            if(isReadActiveOnly) {
113                System.out.println("Reading active region.");
114                System.out.println("source region: "+
115                                   new java.awt.Rectangle(originalSrcMinX,
116                                                          originalSrcMinY,
117                                                          originalSrcWidth,
118                                                          originalSrcHeight));
119                System.out.println("active region: "+
120                                   new java.awt.Rectangle(activeSrcMinX,
121                                                          activeSrcMinY,
122                                                          activeSrcWidth,
123                                                          activeSrcHeight));
124            } else {
125                System.out.println("Reading entire region.");
126                System.out.println("source region: "+
127                                   new java.awt.Rectangle(srcMinX,
128                                                          srcMinY,
129                                                          srcWidth,
130                                                          srcHeight));
131            }
132            System.out.println("destination region: "+
133                               new java.awt.Rectangle(dstMinX,
134                                                      dstMinY,
135                                                      dstWidth,
136                                                      dstHeight));
137        }
138
139        super.beginDecoding();
140    }
141
142    public void decode() throws IOException {
143        super.decode();
144
145        // Reset state.
146        if(isReadActiveOnly) {
147            // Restore original source region values.
148            srcMinX = originalSrcMinX;
149            srcMinY = originalSrcMinY;
150            srcWidth = originalSrcWidth;
151            srcHeight = originalSrcHeight;
152
153            // Unset flag.
154            isReadActiveOnly = false;
155        }
156    }
157
158    public void decodeRaw(byte[] b,
159                          int dstOffset,
160                          int bitsPerPixel,
161                          int scanlineStride) throws IOException {
162        if(isReadActiveOnly) {
163            // Read the active source region only.
164
165            int activeBytesPerRow = (activeSrcWidth*bitsPerPixel + 7)/8;
166            int totalBytesPerRow = (originalSrcWidth*bitsPerPixel + 7)/8;
167            int bytesToSkipPerRow = totalBytesPerRow - activeBytesPerRow;
168
169            //
170            // Seek to the start of the active region:
171            //
172            // active offset = original offset +
173            //                 number of bytes to start of first active row +
174            //                 number of bytes to first active pixel within row
175            //
176            // Since the condition for reading from the active region only is
177            //
178            //     ((activeSrcMinX - srcMinX)*bitsPerPixel) % 8 == 0
179            //
180            // the bit offset to the first active pixel within the first
181            // active row is a multiple of 8.
182            //
183            stream.seek(offset +
184                        (activeSrcMinY - originalSrcMinY)*totalBytesPerRow +
185                        ((activeSrcMinX - originalSrcMinX)*bitsPerPixel)/8);
186
187            int lastRow = activeSrcHeight - 1;
188            for (int y = 0; y < activeSrcHeight; y++) {
189                stream.read(b, dstOffset, activeBytesPerRow);
190                dstOffset += scanlineStride;
191
192                // Skip unneeded bytes (row suffix + row prefix).
193                if(y != lastRow) {
194                    stream.skipBytes(bytesToSkipPerRow);
195                }
196            }
197        } else {
198            // Read the entire source region.
199            stream.seek(offset);
200            int bytesPerRow = (srcWidth*bitsPerPixel + 7)/8;
201            if(bytesPerRow == scanlineStride) {
202                stream.read(b, dstOffset, bytesPerRow*srcHeight);
203            } else {
204                for (int y = 0; y < srcHeight; y++) {
205                    stream.read(b, dstOffset, bytesPerRow);
206                    dstOffset += scanlineStride;
207                }
208            }
209        }
210    }
211}
